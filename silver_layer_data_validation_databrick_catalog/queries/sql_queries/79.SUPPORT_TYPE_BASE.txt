---79) dev_iceberg."tmkn-aws-dwh-dev-iceberg-silver".SUPPORT_TYPE_BASE

WITH bronze_layer AS (
    SELECT
        code,
        supportareaid,
        label,
        is_active,
        ispaymentdetailed,
        issitevisit,
        FALSE AS is_deleted
    FROM dev_iceberg."tmkn-aws-dwh-dev-iceberg-bronze".OSUSR_398_SUPPORTTYPE
),

silver_layer AS (
    SELECT
        code,
        supportarea,
        label,
        is_active,
        ispaymentdetailed,
        issitevisit,
        FALSE AS is_deleted
    FROM dev_iceberg."tmkn-aws-dwh-dev-iceberg-silver".SUPPORT_TYPE_BASE
)

-- =========================
-- VALIDATION
-- =========================

-- COUNT VALIDATION
SELECT 'COUNT_VALIDATION' AS validation_type,
       COUNT(*) AS bronze_count,
       (SELECT COUNT(*) FROM silver_layer) AS silver_count
FROM bronze_layer

UNION ALL

-- DUPLICATE CHECK - BRONZE
SELECT 'DUPLICATE_BRONZE',
       COUNT(*),
       NULL
FROM (
    SELECT code
    FROM bronze_layer
    GROUP BY code
    HAVING COUNT(*) > 1
) d

UNION ALL

-- DUPLICATE CHECK - SILVER
SELECT 'DUPLICATE_SILVER',
       COUNT(*),
       NULL
FROM (
    SELECT code
    FROM silver_layer
    GROUP BY code
    HAVING COUNT(*) > 1
) d

UNION ALL

-- PRIMARY KEY NULL CHECK - BRONZE
SELECT 'PK_NULL_BRONZE',
       COUNT(*),
       NULL
FROM bronze_layer
WHERE code IS NULL

UNION ALL

-- PRIMARY KEY NULL CHECK - SILVER
SELECT 'PK_NULL_SILVER',
       COUNT(*),
       NULL
FROM silver_layer
WHERE code IS NULL

UNION ALL

-- BRONZE NOT IN SILVER
SELECT 'BRONZE_NOT_IN_SILVER',
       COUNT(*),
       NULL
FROM (
    SELECT
        CAST(code AS VARCHAR),
        CAST(supportareaid AS VARCHAR),
        CAST(label AS VARCHAR),
        CAST(is_active AS VARCHAR),
        CAST(ispaymentdetailed AS VARCHAR),
        CAST(issitevisit AS VARCHAR)
    FROM bronze_layer

    EXCEPT

    SELECT
        CAST(code AS VARCHAR),
        CAST(supportarea AS VARCHAR),
        CAST(label AS VARCHAR),
        CAST(is_active AS VARCHAR),
        CAST(ispaymentdetailed AS VARCHAR),
        CAST(issitevisit AS VARCHAR)
    FROM silver_layer
) t

UNION ALL

-- SILVER NOT IN BRONZE
SELECT 'SILVER_NOT_IN_BRONZE',
       COUNT(*),
       NULL
FROM (
    SELECT
        CAST(code AS VARCHAR),
        CAST(supportarea AS VARCHAR),
        CAST(label AS VARCHAR),
        CAST(is_active AS VARCHAR),
        CAST(ispaymentdetailed AS VARCHAR),
        CAST(issitevisit AS VARCHAR)
    FROM silver_layer

    EXCEPT

    SELECT
        CAST(code AS VARCHAR),
        CAST(supportareaid AS VARCHAR),
        CAST(label AS VARCHAR),
        CAST(is_active AS VARCHAR),
        CAST(ispaymentdetailed AS VARCHAR),
        CAST(issitevisit AS VARCHAR)
    FROM bronze_layer
) t

UNION ALL

-- COLUMN MISMATCH COUNT
SELECT 'COLUMN_MISMATCH_COUNT',
       COUNT(*),
       NULL
FROM (
    SELECT b.code
    FROM bronze_layer b
    JOIN silver_layer s
        ON b.code = s.code
    WHERE
        COALESCE(CAST(b.supportareaid AS VARCHAR), '') <> COALESCE(CAST(s.supportarea AS VARCHAR), '')
        OR COALESCE(CAST(b.label AS VARCHAR), '') <> COALESCE(CAST(s.label AS VARCHAR), '')
        OR COALESCE(CAST(b.is_active AS VARCHAR), '') <> COALESCE(CAST(s.is_active AS VARCHAR), '')
        OR COALESCE(CAST(b.ispaymentdetailed AS VARCHAR), '') <> COALESCE(CAST(s.ispaymentdetailed AS VARCHAR), '')
        OR COALESCE(CAST(b.issitevisit AS VARCHAR), '') <> COALESCE(CAST(s.issitevisit AS VARCHAR), '')
) t;
