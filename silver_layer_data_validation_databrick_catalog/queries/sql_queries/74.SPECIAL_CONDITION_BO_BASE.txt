---75) dev_iceberg."tmkn-aws-dwh-dev-iceberg-silver".SPECIAL_CONDITION_BO_BASE

WITH bronze_layer AS (
    SELECT *
    FROM dev_iceberg."tmkn-aws-dwh-dev-iceberg-bronze".OSUSR_T8J_SPECIALCONDITIONBO
),

silver_layer AS (
    SELECT *
    FROM dev_iceberg."tmkn-aws-dwh-dev-iceberg-silver".SPECIAL_CONDITION_BO_BASE
)

-- =========================
-- VALIDATION BLOCK
-- =========================

-- 1) COUNT VALIDATION
SELECT 
    'COUNT_VALIDATION' AS validation_type,
    (SELECT COUNT(*) FROM bronze_layer) AS bronze_count,
    (SELECT COUNT(*) FROM silver_layer) AS silver_count,
    (SELECT COUNT(*) FROM bronze_layer) - (SELECT COUNT(*) FROM silver_layer) AS count_difference

UNION ALL

-- 2) DUPLICATE CHECK - BRONZE
SELECT 
    'DUPLICATE_BRONZE',
    COUNT(*),
    NULL,
    NULL
FROM (
    SELECT id
    FROM bronze_layer
    GROUP BY id
    HAVING COUNT(*) > 1
)

UNION ALL

-- 3) DUPLICATE CHECK - SILVER
SELECT 
    'DUPLICATE_SILVER',
    COUNT(*),
    NULL,
    NULL
FROM (
    SELECT id
    FROM silver_layer
    GROUP BY id
    HAVING COUNT(*) > 1
)

UNION ALL

-- 4) NULL PK CHECK - BRONZE
SELECT 
    'PK_NULL_BRONZE',
    COUNT(*),
    NULL,
    NULL
FROM bronze_layer
WHERE id IS NULL

UNION ALL

-- 5) NULL PK CHECK - SILVER
SELECT 
    'PK_NULL_SILVER',
    COUNT(*),
    NULL,
    NULL
FROM silver_layer
WHERE id IS NULL

UNION ALL

-- 6) BRONZE NOT IN SILVER
SELECT 
    'BRONZE_NOT_IN_SILVER',
    COUNT(*),
    NULL,
    NULL
FROM (
    SELECT 
        CAST(id AS VARCHAR),
        CAST(name AS VARCHAR),
        CAST(description AS VARCHAR)
    FROM bronze_layer

    EXCEPT

    SELECT 
        CAST(id AS VARCHAR),
        CAST(name AS VARCHAR),
        CAST(description AS VARCHAR)
    FROM silver_layer
)

UNION ALL

-- 7) SILVER NOT IN BRONZE
SELECT 
    'SILVER_NOT_IN_BRONZE',
    COUNT(*),
    NULL,
    NULL
FROM (
    SELECT 
        CAST(id AS VARCHAR),
        CAST(name AS VARCHAR),
        CAST(description AS VARCHAR)
    FROM silver_layer

    EXCEPT

    SELECT 
        CAST(id AS VARCHAR),
        CAST(name AS VARCHAR),
        CAST(description AS VARCHAR)
    FROM bronze_layer
)

UNION ALL

-- 8) COLUMN MISMATCH CHECK (SAFE)
SELECT 
    'COLUMN_MISMATCH_COUNT',
    COUNT(*),
    NULL,
    NULL
FROM (
    SELECT *
    FROM bronze_layer b
    JOIN silver_layer s
        ON b.id = s.id
    WHERE 
        COALESCE(CAST(b.name AS VARCHAR),'') 
        IS DISTINCT FROM COALESCE(CAST(s.name AS VARCHAR),'')
     OR COALESCE(CAST(b.description AS VARCHAR),'') 
        IS DISTINCT FROM COALESCE(CAST(s.description AS VARCHAR),'')
     OR COALESCE(CAST(b.isactive AS VARCHAR),'') 
        IS DISTINCT FROM COALESCE(CAST(s.isactive AS VARCHAR),'')
) t;
