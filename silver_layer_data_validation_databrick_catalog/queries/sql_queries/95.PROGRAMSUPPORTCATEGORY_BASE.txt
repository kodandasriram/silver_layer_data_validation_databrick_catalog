---95) dev_iceberg."tmkn-aws-dwh-dev-iceberg-silver".PROGRAMSUPPORTCATEGORY_BASE

WITH bronze_layer AS (
    SELECT
        id,
        CAST(programid AS VARCHAR) AS programid,
        CAST(supportcategoryid AS VARCHAR) AS supportcategoryid,
        CAST(glaccountid AS VARCHAR) AS glaccountid
    FROM dev_iceberg."tmkn-aws-dwh-dev-iceberg-bronze".OSUSR_PX1_PROGRAMSUPPORTCATEGORY
),

silver_layer AS (
    SELECT
        id,
        CAST(programid AS VARCHAR) AS programid,
        CAST(supportcategoryid AS VARCHAR) AS supportcategoryid,
        CAST(glaccountid AS VARCHAR) AS glaccountid
    FROM dev_iceberg."tmkn-aws-dwh-dev-iceberg-silver".PROGRAMSUPPORTCATEGORY_BASE
)

-- =========================
-- VALIDATION
-- =========================

-- COUNT
SELECT 'COUNT_VALIDATION', COUNT(*), (SELECT COUNT(*) FROM silver_layer)
FROM bronze_layer

UNION ALL

-- DUPLICATES
SELECT 'DUPLICATE_BRONZE', COUNT(*), NULL
FROM (
    SELECT id
    FROM bronze_layer
    GROUP BY id
    HAVING COUNT(*) > 1
)

UNION ALL

SELECT 'DUPLICATE_SILVER', COUNT(*), NULL
FROM (
    SELECT id
    FROM silver_layer
    GROUP BY id
    HAVING COUNT(*) > 1
)

UNION ALL

-- PK NULL
SELECT 'PK_NULL_BRONZE', COUNT(*), NULL
FROM bronze_layer
WHERE id IS NULL

UNION ALL

SELECT 'PK_NULL_SILVER', COUNT(*), NULL
FROM silver_layer
WHERE id IS NULL

UNION ALL

-- BRONZE NOT IN SILVER
SELECT 'BRONZE_NOT_IN_SILVER', COUNT(*), NULL
FROM (
    SELECT id, programid, supportcategoryid, glaccountid
    FROM bronze_layer

    EXCEPT

    SELECT id, programid, supportcategoryid, glaccountid
    FROM silver_layer
)

UNION ALL

-- SILVER NOT IN BRONZE
SELECT 'SILVER_NOT_IN_BRONZE', COUNT(*), NULL
FROM (
    SELECT id, programid, supportcategoryid, glaccountid
    FROM silver_layer

    EXCEPT

    SELECT id, programid, supportcategoryid, glaccountid
    FROM bronze_layer
)

UNION ALL

-- COLUMN MISMATCH
SELECT 'COLUMN_MISMATCH_COUNT', COUNT(*), NULL
FROM (
    SELECT *
    FROM bronze_layer b
    JOIN silver_layer s
        ON b.id = s.id
    WHERE
        COALESCE(b.programid,'') <> COALESCE(s.programid,'')
     OR COALESCE(b.supportcategoryid,'') <> COALESCE(s.supportcategoryid,'')
     OR COALESCE(b.glaccountid,'') <> COALESCE(s.glaccountid,'')
) t;
