---96) dev_iceberg."tmkn-aws-dwh-dev-iceberg-silver".GLACCOUNT_BASE

WITH bronze_layer AS (
    SELECT
        id,
        name,
        number,
        status,
        "index"
    FROM dev_iceberg."tmkn-aws-dwh-dev-iceberg-bronze".OSUSR_PX1_GLACCOUNT
),

silver_layer AS (
    SELECT
        id,
        name,
        number,
        status,
        "index"
    FROM dev_iceberg."tmkn-aws-dwh-dev-iceberg-silver".GLACCOUNT_BASE
)

-- =========================
-- VALIDATION
-- =========================

-- COUNT
SELECT 'COUNT_VALIDATION', COUNT(*), (SELECT COUNT(*) FROM silver_layer)
FROM bronze_layer

UNION ALL

-- DUPLICATES (BRONZE)
SELECT 'DUPLICATE_BRONZE', COUNT(*), NULL
FROM (
    SELECT id
    FROM bronze_layer
    GROUP BY id
    HAVING COUNT(*) > 1
)

UNION ALL

-- DUPLICATES (SILVER)
SELECT 'DUPLICATE_SILVER', COUNT(*), NULL
FROM (
    SELECT id
    FROM silver_layer
    GROUP BY id
    HAVING COUNT(*) > 1
)

UNION ALL

-- PK NULL (BRONZE)
SELECT 'PK_NULL_BRONZE', COUNT(*), NULL
FROM bronze_layer
WHERE id IS NULL

UNION ALL

-- PK NULL (SILVER)
SELECT 'PK_NULL_SILVER', COUNT(*), NULL
FROM silver_layer
WHERE id IS NULL

UNION ALL

-- BRONZE NOT IN SILVER
SELECT 'BRONZE_NOT_IN_SILVER', COUNT(*), NULL
FROM (
    SELECT
        CAST(id AS VARCHAR),
        CAST(name AS VARCHAR),
        CAST(number AS VARCHAR),
        CAST(status AS VARCHAR),
        CAST("index" AS VARCHAR)
    FROM bronze_layer

    EXCEPT

    SELECT
        CAST(id AS VARCHAR),
        CAST(name AS VARCHAR),
        CAST(number AS VARCHAR),
        CAST(status AS VARCHAR),
        CAST("index" AS VARCHAR)
    FROM silver_layer
)

UNION ALL

-- SILVER NOT IN BRONZE
SELECT 'SILVER_NOT_IN_BRONZE', COUNT(*), NULL
FROM (
    SELECT
        CAST(id AS VARCHAR),
        CAST(name AS VARCHAR),
        CAST(number AS VARCHAR),
        CAST(status AS VARCHAR),
        CAST("index" AS VARCHAR)
    FROM silver_layer

    EXCEPT

    SELECT
        CAST(id AS VARCHAR),
        CAST(name AS VARCHAR),
        CAST(number AS VARCHAR),
        CAST(status AS VARCHAR),
        CAST("index" AS VARCHAR)
    FROM bronze_layer
)

UNION ALL

-- COLUMN MISMATCH COUNT
SELECT 'COLUMN_MISMATCH_COUNT', COUNT(*), NULL
FROM (
    SELECT *
    FROM bronze_layer b
    JOIN silver_layer s
        ON b.id = s.id
    WHERE
           COALESCE(CAST(b.name AS VARCHAR),'')   <> COALESCE(CAST(s.name AS VARCHAR),'')
        OR COALESCE(CAST(b.number AS VARCHAR),'') <> COALESCE(CAST(s.number AS VARCHAR),'')
        OR COALESCE(CAST(b.status AS VARCHAR),'') <> COALESCE(CAST(s.status AS VARCHAR),'')
        OR COALESCE(CAST(b."index" AS VARCHAR),'')<> COALESCE(CAST(s."index" AS VARCHAR),'')
) t;
