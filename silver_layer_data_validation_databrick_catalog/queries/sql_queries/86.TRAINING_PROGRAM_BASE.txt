---86) dev_iceberg."tmkn-aws-dwh-dev-iceberg-silver".TRAINING_PROGRAM_BASE

WITH bronze_layer AS (
    SELECT
        COALESCE(r.id, p.id) AS id,
        r.authorizedproviderid,
        COALESCE(r.name, p.trainingprogramname) AS name,
        COALESCE(r.overview, p.overview) AS overview,
        COALESCE(r.awardingbodyname, p.awardingbody) AS awardingbodyname,
        r.trainingprogramtypeid,
        COALESCE(r.trainingknowledgeareaid, p.knowledgeareaid) AS knowledgeareaid,
        r.trainingdetailareaid,
        r.traininghours,
        COALESCE(r.estimatedtrainingcost, p.totalprice) AS estimatedtrainingcost,
        COALESCE(
            p.rateperhour,
            CASE
                WHEN r.traininghours <> 0
                THEN r.estimatedtrainingcost / r.traininghours
            END
        ) AS rateperhour,
        p.trainingprovider,
        p.modeofstudy,
        p.nvqlevelofotherlevels,
        FALSE AS is_deleted
    FROM dev_iceberg."tmkn-aws-dwh-dev-iceberg-bronze".OSUSR_R9T_TRAININGPROGRAM r
    FULL OUTER JOIN dev_iceberg."tmkn-aws-dwh-dev-iceberg-bronze".OSUSR_PX1_TRAININGPROGRAM p
        ON r.id = p.id
),

silver_layer AS (
    SELECT
        id,
        authorizedproviderid,
        name,
        overview,
        awardingbodyname,
        trainingprogramtypeid,
        knowledgeareaid,
        trainingdetailareaid,
        traininghours,
        estimatedtrainingcost,
        rateperhour,
        trainingprovider,
        modeofstudy,
        nvqlevelofotherlevels,
        COALESCE(is_deleted, FALSE) AS is_deleted
    FROM dev_iceberg."tmkn-aws-dwh-dev-iceberg-silver".TRAINING_PROGRAM_BASE
)

-- =========================
-- VALIDATION
-- =========================

SELECT 'COUNT_VALIDATION', COUNT(*), (SELECT COUNT(*) FROM silver_layer)
FROM bronze_layer

UNION ALL

SELECT 'DUPLICATE_BRONZE', COUNT(*), NULL
FROM (
    SELECT id
    FROM bronze_layer
    GROUP BY id
    HAVING COUNT(*) > 1
) t

UNION ALL

SELECT 'DUPLICATE_SILVER', COUNT(*), NULL
FROM (
    SELECT id
    FROM silver_layer
    GROUP BY id
    HAVING COUNT(*) > 1
) t

UNION ALL

SELECT 'PK_NULL_BRONZE', COUNT(*), NULL
FROM bronze_layer
WHERE id IS NULL

UNION ALL

SELECT 'PK_NULL_SILVER', COUNT(*), NULL
FROM silver_layer
WHERE id IS NULL

UNION ALL

SELECT 'BRONZE_NOT_IN_SILVER', COUNT(*), NULL
FROM (
    SELECT
        CAST(id AS VARCHAR),
        CAST(name AS VARCHAR),
        CAST(estimatedtrainingcost AS VARCHAR)
    FROM bronze_layer

    EXCEPT

    SELECT
        CAST(id AS VARCHAR),
        CAST(name AS VARCHAR),
        CAST(estimatedtrainingcost AS VARCHAR)
    FROM silver_layer
) t

UNION ALL

SELECT 'SILVER_NOT_IN_BRONZE', COUNT(*), NULL
FROM (
    SELECT
        CAST(id AS VARCHAR),
        CAST(name AS VARCHAR),
        CAST(estimatedtrainingcost AS VARCHAR)
    FROM silver_layer

    EXCEPT

    SELECT
        CAST(id AS VARCHAR),
        CAST(name AS VARCHAR),
        CAST(estimatedtrainingcost AS VARCHAR)
    FROM bronze_layer
) t

UNION ALL

SELECT 'COLUMN_MISMATCH_COUNT', COUNT(*), NULL
FROM (
    SELECT b.id
    FROM bronze_layer b
    JOIN silver_layer s
        ON b.id = s.id
    WHERE
        COALESCE(CAST(b.name AS VARCHAR), '') <> COALESCE(CAST(s.name AS VARCHAR), '')
        OR COALESCE(CAST(b.overview AS VARCHAR), '') <> COALESCE(CAST(s.overview AS VARCHAR), '')
        OR COALESCE(CAST(b.awardingbodyname AS VARCHAR), '') <> COALESCE(CAST(s.awardingbodyname AS VARCHAR), '')
        OR COALESCE(CAST(b.trainingprogramtypeid AS VARCHAR), '') <> COALESCE(CAST(s.trainingprogramtypeid AS VARCHAR), '')
        OR COALESCE(CAST(b.knowledgeareaid AS VARCHAR), '') <> COALESCE(CAST(s.knowledgeareaid AS VARCHAR), '')
        OR COALESCE(CAST(b.estimatedtrainingcost AS VARCHAR), '') <> COALESCE(CAST(s.estimatedtrainingcost AS VARCHAR), '')
        OR COALESCE(CAST(b.rateperhour AS VARCHAR), '') <> COALESCE(CAST(s.rateperhour AS VARCHAR), '')
        OR COALESCE(CAST(b.trainingprovider AS VARCHAR), '') <> COALESCE(CAST(s.trainingprovider AS VARCHAR), '')
        OR COALESCE(CAST(b.modeofstudy AS VARCHAR), '') <> COALESCE(CAST(s.modeofstudy AS VARCHAR), '')
        OR COALESCE(CAST(b.nvqlevelofotherlevels AS VARCHAR), '') <> COALESCE(CAST(s.nvqlevelofotherlevels AS VARCHAR), '')
) t;
