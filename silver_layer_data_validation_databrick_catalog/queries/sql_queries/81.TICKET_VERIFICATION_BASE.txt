WITH bronze_layer AS (
    SELECT
        id,
        ticketid,
        verificationmoduleid,
        verificationstatusid,
        haserror,
        createdon,
        updatedon,
        FALSE AS is_deleted
    FROM dev_iceberg."tmkn-aws-dwh-dev-iceberg-bronze".OSUSR_5HX_TICKETVERIFICATION
),

silver_layer AS (
    SELECT
        id,
        ticketid,
        verificationmoduleid,
        verificationstatusid,
        haserror,
        createdon,
        updatedon,
        FALSE AS is_deleted
    FROM dev_iceberg."tmkn-aws-dwh-dev-iceberg-silver".TICKET_VERIFICATION_BASE
)

-- =========================
-- VALIDATION
-- =========================

SELECT 'COUNT_VALIDATION', COUNT(*), (SELECT COUNT(*) FROM silver_layer)
FROM bronze_layer

UNION ALL

SELECT 'DUPLICATE_BRONZE', COUNT(*), NULL
FROM (
    SELECT id
    FROM bronze_layer
    GROUP BY id
    HAVING COUNT(*) > 1
) t

UNION ALL

SELECT 'DUPLICATE_SILVER', COUNT(*), NULL
FROM (
    SELECT id
    FROM silver_layer
    GROUP BY id
    HAVING COUNT(*) > 1
) t

UNION ALL

SELECT 'PK_NULL_BRONZE', COUNT(*), NULL
FROM bronze_layer
WHERE id IS NULL

UNION ALL

SELECT 'PK_NULL_SILVER', COUNT(*), NULL
FROM silver_layer
WHERE id IS NULL

UNION ALL

SELECT 'BRONZE_NOT_IN_SILVER', COUNT(*), NULL
FROM (
    SELECT
        CAST(id AS VARCHAR),
        CAST(ticketid AS VARCHAR),
        CAST(verificationstatusid AS VARCHAR)
    FROM bronze_layer

    EXCEPT

    SELECT
        CAST(id AS VARCHAR),
        CAST(ticketid AS VARCHAR),
        CAST(verificationstatusid AS VARCHAR)
    FROM silver_layer
) t

UNION ALL

SELECT 'SILVER_NOT_IN_BRONZE', COUNT(*), NULL
FROM (
    SELECT
        CAST(id AS VARCHAR),
        CAST(ticketid AS VARCHAR),
        CAST(verificationstatusid AS VARCHAR)
    FROM silver_layer

    EXCEPT

    SELECT
        CAST(id AS VARCHAR),
        CAST(ticketid AS VARCHAR),
        CAST(verificationstatusid AS VARCHAR)
    FROM bronze_layer
) t

UNION ALL

SELECT 'COLUMN_MISMATCH_COUNT', COUNT(*), NULL
FROM (
    SELECT b.id
    FROM bronze_layer b
    JOIN silver_layer s
        ON b.id = s.id
    WHERE
        COALESCE(CAST(b.ticketid AS VARCHAR), '') <> COALESCE(CAST(s.ticketid AS VARCHAR), '')
        OR COALESCE(CAST(b.verificationmoduleid AS VARCHAR), '') <> COALESCE(CAST(s.verificationmoduleid AS VARCHAR), '')
        OR COALESCE(CAST(b.verificationstatusid AS VARCHAR), '') <> COALESCE(CAST(s.verificationstatusid AS VARCHAR), '')
        OR COALESCE(CAST(b.haserror AS VARCHAR), '') <> COALESCE(CAST(s.haserror AS VARCHAR), '')
        OR COALESCE(CAST(b.createdon AS VARCHAR), '') <> COALESCE(CAST(s.createdon AS VARCHAR), '')
        OR COALESCE(CAST(b.updatedon AS VARCHAR), '') <> COALESCE(CAST(s.updatedon AS VARCHAR), '')
) t;
