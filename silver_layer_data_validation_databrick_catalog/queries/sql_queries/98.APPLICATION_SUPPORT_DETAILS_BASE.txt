-- =========================

-- =========================

-- COUNT
SELECT 'COUNT_VALIDATION', COUNT(*), (SELECT COUNT(*) FROM silver_layer)
FROM bronze_layer

UNION ALL

-- DUPLICATES BRONZE
SELECT 'DUPLICATE_BRONZE', COUNT(*), NULL
FROM (
    SELECT id
    FROM bronze_layer
    GROUP BY id
    HAVING COUNT(*) > 1
)

UNION ALL

-- DUPLICATES SILVER
SELECT 'DUPLICATE_SILVER', COUNT(*), NULL
FROM (
    SELECT id
    FROM silver_layer
    GROUP BY id
    HAVING COUNT(*) > 1
)

UNION ALL

-- PK NULL BRONZE
SELECT 'PK_NULL_BRONZE', COUNT(*), NULL
FROM bronze_layer
WHERE id IS NULL

UNION ALL

-- PK NULL SILVER
SELECT 'PK_NULL_SILVER', COUNT(*), NULL
FROM silver_layer
WHERE id IS NULL

UNION ALL

-- BRONZE NOT IN SILVER
SELECT 'BRONZE_NOT_IN_SILVER', COUNT(*), NULL
FROM (
    SELECT
        CAST(id AS VARCHAR),
        CAST(itemname AS VARCHAR),
        CAST(totalcost AS VARCHAR)
    FROM bronze_layer

    EXCEPT

    SELECT
        id,
        CAST(itemname AS VARCHAR),
        CAST(totalcost AS VARCHAR)
    FROM silver_layer
)

UNION ALL

-- SILVER NOT IN BRONZE
SELECT 'SILVER_NOT_IN_BRONZE', COUNT(*), NULL
FROM (
    SELECT
        id,
        CAST(itemname AS VARCHAR),
        CAST(totalcost AS VARCHAR)
    FROM silver_layer

    EXCEPT

    SELECT
        CAST(id AS VARCHAR),
        CAST(itemname AS VARCHAR),
        CAST(totalcost AS VARCHAR)
    FROM bronze_layer
)

UNION ALL

-- COLUMN MISMATCH (LIGHT CHECK)
SELECT 'COLUMN_MISMATCH_COUNT', COUNT(*), NULL
FROM (
    SELECT *
    FROM bronze_layer b
    JOIN silver_layer s
        ON CAST(b.id AS VARCHAR) = s.id
    WHERE
           COALESCE(CAST(b.itemreference AS VARCHAR),'') <> COALESCE(CAST(s.itemreference AS VARCHAR),'')
        OR COALESCE(CAST(b.itemid AS VARCHAR),'') <> COALESCE(CAST(s.itemid AS VARCHAR),'')
