---87) dev_iceberg."tmkn-aws-dwh-dev-iceberg-silver".TRAININGTRACK_BASE

WITH bronze_layer AS (
    SELECT
        id,
        label,
        FALSE AS is_deleted
    FROM dev_iceberg."tmkn-aws-dwh-dev-iceberg-bronze".OSUSR_3QQ_TRAININGTRACK
),

silver_layer AS (
    SELECT
        id,
        label,
        FALSE AS is_deleted
    FROM dev_iceberg."tmkn-aws-dwh-dev-iceberg-silver".TRAININGTRACK_BASE
)

-- =========================
-- VALIDATION
-- =========================

SELECT 'COUNT_VALIDATION', COUNT(*), (SELECT COUNT(*) FROM silver_layer)
FROM bronze_layer

UNION ALL

SELECT 'DUPLICATE_BRONZE', COUNT(*), NULL
FROM (
    SELECT id
    FROM bronze_layer
    GROUP BY id
    HAVING COUNT(*) > 1
) t

UNION ALL

SELECT 'DUPLICATE_SILVER', COUNT(*), NULL
FROM (
    SELECT id
    FROM silver_layer
    GROUP BY id
    HAVING COUNT(*) > 1
) t

UNION ALL

SELECT 'PK_NULL_BRONZE', COUNT(*), NULL
FROM bronze_layer
WHERE id IS NULL

UNION ALL

SELECT 'PK_NULL_SILVER', COUNT(*), NULL
FROM silver_layer
WHERE id IS NULL

UNION ALL

SELECT 'BRONZE_NOT_IN_SILVER', COUNT(*), NULL
FROM (
    SELECT
        CAST(id AS VARCHAR),
        CAST(label AS VARCHAR)
    FROM bronze_layer

    EXCEPT

    SELECT
        CAST(id AS VARCHAR),
        CAST(label AS VARCHAR)
    FROM silver_layer
) t

UNION ALL

SELECT 'SILVER_NOT_IN_BRONZE', COUNT(*), NULL
FROM (
    SELECT
        CAST(id AS VARCHAR),
        CAST(label AS VARCHAR)
    FROM silver_layer

    EXCEPT

    SELECT
        CAST(id AS VARCHAR),
        CAST(label AS VARCHAR)
    FROM bronze_layer
) t

UNION ALL

SELECT 'COLUMN_MISMATCH_COUNT', COUNT(*), NULL
FROM (
    SELECT b.id
    FROM bronze_layer b
    JOIN silver_layer s
        ON b.id = s.id
    WHERE
        COALESCE(CAST(b.label AS VARCHAR), '') <> COALESCE(CAST(s.label AS VARCHAR), '')
) t;
