WITH bronze_layer AS (
    SELECT
        applicationsupportid,
        itemcapamt,
        certificationid,
        trainingstartdate,
        trainingenddate,
        trainingassessmentdate,
        trainingdeliverytypeid,
        trainingpaymenttypeid,
        itemvatamt,
        itemamtavailable,
        itemamtinprogress,
        customersharetotal,
        tkshareactualpct,
        tkshareamtauto,
        tkshareamt,
        customershare,
        payeetypeid,
        FALSE AS is_deleted
    FROM dev_iceberg."tmkn-aws-dwh-dev-iceberg-bronze".OSUSR_VW9_TRAINING
),

silver_layer AS (
    SELECT
        applicationsupportid,
        itemcapamt,
        certificationid,
        trainingstartdate,
        trainingenddate,
        trainingassessmentdate,
        trainingdeliverytypeid,
        trainingpaymenttypeid,
        itemvatamt,
        itemamtavailable,
        itemamtinprogress,
        customersharetotal,
        tkshareactualpct,
        tkshareamtauto,
        tkshareamt,
        customershare,
        payeetypeid,
        COALESCE(is_deleted, FALSE) AS is_deleted
    FROM dev_iceberg."tmkn-aws-dwh-dev-iceberg-silver".TRAINING_BASE
)

-- =========================
-- VALIDATION
-- =========================

SELECT 'COUNT_VALIDATION', COUNT(*), (SELECT COUNT(*) FROM silver_layer)
FROM bronze_layer

UNION ALL

SELECT 'DUPLICATE_BRONZE', COUNT(*), NULL
FROM (
    SELECT applicationsupportid
    FROM bronze_layer
    GROUP BY applicationsupportid
    HAVING COUNT(*) > 1
) t

UNION ALL

SELECT 'DUPLICATE_SILVER', COUNT(*), NULL
FROM (
    SELECT applicationsupportid
    FROM silver_layer
    GROUP BY applicationsupportid
    HAVING COUNT(*) > 1
) t

UNION ALL

SELECT 'PK_NULL_BRONZE', COUNT(*), NULL
FROM bronze_layer
WHERE applicationsupportid IS NULL

UNION ALL

SELECT 'PK_NULL_SILVER', COUNT(*), NULL
FROM silver_layer
WHERE applicationsupportid IS NULL

UNION ALL

SELECT 'BRONZE_NOT_IN_SILVER', COUNT(*), NULL
FROM (
    SELECT
        CAST(applicationsupportid AS VARCHAR),
        CAST(certificationid AS VARCHAR),
        CAST(itemcapamt AS VARCHAR)
    FROM bronze_layer

    EXCEPT

    SELECT
        CAST(applicationsupportid AS VARCHAR),
        CAST(certificationid AS VARCHAR),
        CAST(itemcapamt AS VARCHAR)
    FROM silver_layer
) t

UNION ALL

SELECT 'SILVER_NOT_IN_BRONZE', COUNT(*), NULL
FROM (
    SELECT
        CAST(applicationsupportid AS VARCHAR),
        CAST(certificationid AS VARCHAR),
        CAST(itemcapamt AS VARCHAR)
    FROM silver_layer

    EXCEPT

    SELECT
        CAST(applicationsupportid AS VARCHAR),
        CAST(certificationid AS VARCHAR),
        CAST(itemcapamt AS VARCHAR)
    FROM bronze_layer
) t

UNION ALL

SELECT 'COLUMN_MISMATCH_COUNT', COUNT(*), NULL
FROM (
    SELECT b.applicationsupportid
    FROM bronze_layer b
    JOIN silver_layer s
        ON b.applicationsupportid = s.applicationsupportid
    WHERE
        COALESCE(CAST(b.itemcapamt AS VARCHAR), '') <> COALESCE(CAST(s.itemcapamt AS VARCHAR), '')
        OR COALESCE(CAST(b.certificationid AS VARCHAR), '') <> COALESCE(CAST(s.certificationid AS VARCHAR), '')
        OR COALESCE(CAST(b.trainingstartdate AS VARCHAR), '') <> COALESCE(CAST(s.trainingstartdate AS VARCHAR), '')
        OR COALESCE(CAST(b.trainingenddate AS VARCHAR), '') <> COALESCE(CAST(s.trainingenddate AS VARCHAR), '')
        OR COALESCE(CAST(b.trainingassessmentdate AS VARCHAR), '') <> COALESCE(CAST(s.trainingassessmentdate AS VARCHAR), '')
        OR COALESCE(CAST(b.trainingdeliverytypeid AS VARCHAR), '') <> COALESCE(CAST(s.trainingdeliverytypeid AS VARCHAR), '')
        OR COALESCE(CAST(b.trainingpaymenttypeid AS VARCHAR), '') <> COALESCE(CAST(s.trainingpaymenttypeid AS VARCHAR), '')
        OR COALESCE(CAST(b.itemvatamt AS VARCHAR), '') <> COALESCE(CAST(s.itemvatamt AS VARCHAR), '')
        OR COALESCE(CAST(b.itemamtavailable AS VARCHAR), '') <> COALESCE(CAST(s.itemamtavailable AS VARCHAR), '')
        OR COALESCE(CAST(b.itemamtinprogress AS VARCHAR), '') <> COALESCE(CAST(s.itemamtinprogress AS VARCHAR), '')
        OR COALESCE(CAST(b.customersharetotal AS VARCHAR), '') <> COALESCE(CAST(s.customersharetotal AS VARCHAR), '')
        OR COALESCE(CAST(b.tkshareactualpct AS VARCHAR), '') <> COALESCE(CAST(s.tkshareactualpct AS VARCHAR), '')
        OR COALESCE(CAST(b.tkshareamtauto AS VARCHAR), '') <> COALESCE(CAST(s.tkshareamtauto AS VARCHAR), '')
        OR COALESCE(CAST(b.tkshareamt AS VARCHAR), '') <> COALESCE(CAST(s.tkshareamt AS VARCHAR), '')
        OR COALESCE(CAST(b.customershare AS VARCHAR), '') <> COALESCE(CAST(s.customershare AS VARCHAR), '')
        OR COALESCE(CAST(b.payeetypeid AS VARCHAR), '') <> COALESCE(CAST(s.payeetypeid AS VARCHAR), '')
) t;
