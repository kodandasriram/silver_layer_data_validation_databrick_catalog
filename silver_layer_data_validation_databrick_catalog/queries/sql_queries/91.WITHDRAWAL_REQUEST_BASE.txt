---91) dev_iceberg."tmkn-aws-dwh-dev-iceberg-silver".WITHDRAWAL_REQUEST_BASE

WITH bronze_layer AS (
SELECT
id,
processid,
statusid,
applicationid,
initiatoruserid,
referencenumber,
startdate,
remarks,
tkremarks,
attribute1,
FALSE AS is_deleted
FROM dev_iceberg."tmkn-aws-dwh-dev-iceberg-bronze".OSUSR_NTP_WITHDRAWALREQUEST4
),

silver_layer AS (
SELECT
id,
processid,
statusid,
applicationid,
initiatoruserid,
referencenumber,
startdate,
remarks,
tkremarks,
attribute1,
is_deleted
FROM dev_iceberg."tmkn-aws-dwh-dev-iceberg-silver".WITHDRAWAL_REQUEST_BASE
)

-- =========================
-- VALIDATION
-- =========================

SELECT 'COUNT_VALIDATION' AS validation_type,
COUNT(*) AS bronze_count,
(SELECT COUNT(*) FROM silver_layer) AS silver_count
FROM bronze_layer

UNION ALL

SELECT 'DUPLICATE_BRONZE', COUNT(*), NULL
FROM (
SELECT id
FROM bronze_layer
GROUP BY id
HAVING COUNT(*) > 1
) b

UNION ALL

SELECT 'DUPLICATE_SILVER', COUNT(*), NULL
FROM (
SELECT id
FROM silver_layer
GROUP BY id
HAVING COUNT(*) > 1
) s

UNION ALL

SELECT 'PK_NULL_BRONZE', COUNT(*), NULL
FROM bronze_layer
WHERE id IS NULL

UNION ALL

SELECT 'PK_NULL_SILVER', COUNT(*), NULL
FROM silver_layer
WHERE id IS NULL

UNION ALL

SELECT 'BRONZE_NOT_IN_SILVER', COUNT(*), NULL
FROM (
SELECT
CAST(id AS VARCHAR),
CAST(referencenumber AS VARCHAR),
CAST(statusid AS VARCHAR)
FROM bronze_layer

EXCEPT

SELECT
    CAST(id AS VARCHAR),
    CAST(referencenumber AS VARCHAR),
    CAST(statusid AS VARCHAR)
FROM silver_layer


) bns

UNION ALL

SELECT 'SILVER_NOT_IN_BRONZE', COUNT(*), NULL
FROM (
SELECT
CAST(id AS VARCHAR),
CAST(referencenumber AS VARCHAR),
CAST(statusid AS VARCHAR)
FROM silver_layer


EXCEPT

SELECT
    CAST(id AS VARCHAR),
    CAST(referencenumber AS VARCHAR),
    CAST(statusid AS VARCHAR)
FROM bronze_layer


) snb

UNION ALL

SELECT 'COLUMN_MISMATCH_COUNT', COUNT(*), NULL
FROM (
SELECT b.id
FROM bronze_layer b
JOIN silver_layer s
ON CAST(b.id AS VARCHAR) = CAST(s.id AS VARCHAR)
WHERE COALESCE(CAST(b.processid AS VARCHAR), '') <> COALESCE(CAST(s.processid AS VARCHAR), '')
OR COALESCE(CAST(b.statusid AS VARCHAR), '') <> COALESCE(CAST(s.statusid AS VARCHAR), '')
OR COALESCE(CAST(b.applicationid AS VARCHAR), '') <> COALESCE(CAST(s.applicationid AS VARCHAR), '')
OR COALESCE(CAST(b.initiatoruserid AS VARCHAR), '') <> COALESCE(CAST(s.initiatoruserid AS VARCHAR), '')
OR COALESCE(CAST(b.referencenumber AS VARCHAR), '') <> COALESCE(CAST(s.referencenumber AS VARCHAR), '')
OR COALESCE(CAST(b.startdate AS VARCHAR), '') <> COALESCE(CAST(s.startdate AS VARCHAR), '')
OR COALESCE(CAST(b.remarks AS VARCHAR), '') <> COALESCE(CAST(s.remarks AS VARCHAR), '')
OR COALESCE(CAST(b.tkremarks AS VARCHAR), '') <> COALESCE(CAST(s.tkremarks AS VARCHAR), '')
OR COALESCE(CAST(b.attribute1 AS VARCHAR), '') <> COALESCE(CAST(s.attribute1 AS VARCHAR), '')
) t;
