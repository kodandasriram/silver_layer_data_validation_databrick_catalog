        s.lastname,
        s.cpr,

        s.posesprofessionalcertificate,
        s.approvedon

    FROM dev_iceberg."tmkn-aws-dwh-dev-iceberg-bronze".OSUSR_5W2_APPLICATIONEMPLOYEE s

silver_layer AS (

    SELECT
        CAST(id AS VARCHAR) AS id,

        applicationid,
        applicationanswer,
        firstname,
        middlename,
        lastname,
        cpr,
        citizenship,
        gender,
        dateofbirth,
        primaryemail,

        contactnumbercountrycode,
        contactnumber,
        secondarycontactcountrycode,
        secondarycontact,

        addressdetailflat,
        addressdetailbuilding,
        addressdetailroad,
        addressdetailblock,
        addressdetailarea,

        academicdegree,
        graduationdate,
        schooluniversityspecial,

        status,

        recommendedgrantbyrm,
        recommendedgrantbyassessor,
        approvedgrant,
        posesprofessionalcertificate,
        approvedon

    FROM dev_iceberg."tmkn-aws-dwh-dev-iceberg-silver".APPLICATIONEMPLOYEE_BASE
)

-- =========================
-- VALIDATION SECTION
-- =========================

-- COUNT CHECK
SELECT 'COUNT_VALIDATION', COUNT(*), (SELECT COUNT(*) FROM silver_layer)
FROM bronze_layer

UNION ALL

-- DUPLICATE CHECK BRONZE
SELECT 'DUPLICATE_BRONZE', COUNT(*), NULL
FROM (
    SELECT id
    FROM bronze_layer
    GROUP BY id
    HAVING COUNT(*) > 1
)

UNION ALL

-- DUPLICATE CHECK SILVER
SELECT 'DUPLICATE_SILVER', COUNT(*), NULL
FROM (
    SELECT id
    FROM silver_layer
    GROUP BY id
    HAVING COUNT(*) > 1
)

UNION ALL

-- PK NULL BRONZE
SELECT 'PK_NULL_BRONZE', COUNT(*), NULL
FROM bronze_layer
WHERE id IS NULL

UNION ALL

-- PK NULL SILVER
SELECT 'PK_NULL_SILVER', COUNT(*), NULL
FROM silver_layer
WHERE id IS NULL

UNION ALL

-- BRONZE NOT IN SILVER
SELECT 'BRONZE_NOT_IN_SILVER', COUNT(*), NULL
FROM (
    SELECT
        CAST(id AS VARCHAR),
        CAST(firstname AS VARCHAR),
        CAST(lastname AS VARCHAR),
        CAST(primaryemail AS VARCHAR)
    FROM bronze_layer

    EXCEPT

    SELECT
        id,
        CAST(firstname AS VARCHAR),
        CAST(lastname AS VARCHAR),
        CAST(primaryemail AS VARCHAR)
    FROM silver_layer
)

UNION ALL

-- SILVER NOT IN BRONZE
SELECT 'SILVER_NOT_IN_BRONZE', COUNT(*), NULL
FROM (
    SELECT
        id,
        CAST(firstname AS VARCHAR),
        CAST(lastname AS VARCHAR),
        CAST(primaryemail AS VARCHAR)
    FROM silver_layer

    EXCEPT

    SELECT
        CAST(id AS VARCHAR),
        CAST(firstname AS VARCHAR),
        CAST(lastname AS VARCHAR),
        CAST(primaryemail AS VARCHAR)
    FROM bronze_layer
)

UNION ALL

-- COLUMN MISMATCH CHECK
