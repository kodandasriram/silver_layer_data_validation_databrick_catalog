WITH bronze_layer AS (
    SELECT
        id,
        processid,
        tickettypeid,
        ticketstatusid,
        entityidentifier,
        refnumber,
        closedon,
        isactive,
        insertedpassword,
        tkchannelid,
        FALSE AS is_deleted
    FROM dev_iceberg."tmkn-aws-dwh-dev-iceberg-bronze".OSUSR_5HX_TICKET
),

silver_layer AS (
    SELECT
        id,
        processid,
        tickettype,
        ticketstatus,
        entityidentifier,
        refnumber,
        closedon,
        isactive,
        insertedpassword,
        tkchannel,
        is_deleted
    FROM dev_iceberg."tmkn-aws-dwh-dev-iceberg-silver".TICKET_BASE
)

-- =========================
-- VALIDATION
-- =========================

-- COUNT VALIDATION
SELECT 'COUNT_VALIDATION', COUNT(*), (SELECT COUNT(*) FROM silver_layer)
FROM bronze_layer

UNION ALL

-- DUPLICATE CHECK - BRONZE
SELECT 'DUPLICATE_BRONZE', COUNT(*), NULL
FROM (
    SELECT id
    FROM bronze_layer
    GROUP BY id
    HAVING COUNT(*) > 1
)

UNION ALL

-- DUPLICATE CHECK - SILVER
SELECT 'DUPLICATE_SILVER', COUNT(*), NULL
FROM (
    SELECT id
    FROM silver_layer
    GROUP BY id
    HAVING COUNT(*) > 1
)

UNION ALL

-- PRIMARY KEY NULL CHECK - BRONZE
SELECT 'PK_NULL_BRONZE', COUNT(*), NULL
FROM bronze_layer
WHERE id IS NULL

UNION ALL

-- PRIMARY KEY NULL CHECK - SILVER
SELECT 'PK_NULL_SILVER', COUNT(*), NULL
FROM silver_layer
WHERE id IS NULL

UNION ALL

-- BRONZE NOT IN SILVER
SELECT 'BRONZE_NOT_IN_SILVER', COUNT(*), NULL
FROM (
    SELECT
        CAST(id AS VARCHAR),
        CAST(processid AS VARCHAR),
        CAST(refnumber AS VARCHAR)
    FROM bronze_layer

    EXCEPT

    SELECT
        CAST(id AS VARCHAR),
        CAST(processid AS VARCHAR),
        CAST(refnumber AS VARCHAR)
    FROM silver_layer
) t

UNION ALL

-- SILVER NOT IN BRONZE
SELECT 'SILVER_NOT_IN_BRONZE', COUNT(*), NULL
FROM (
    SELECT
        CAST(id AS VARCHAR),
        CAST(processid AS VARCHAR),
        CAST(refnumber AS VARCHAR)
    FROM silver_layer

    EXCEPT

    SELECT
        CAST(id AS VARCHAR),
        CAST(processid AS VARCHAR),
        CAST(refnumber AS VARCHAR)
    FROM bronze_layer
) t

UNION ALL

-- COLUMN MISMATCH COUNT
SELECT 'COLUMN_MISMATCH_COUNT', COUNT(*), NULL
FROM (
    SELECT b.id
    FROM bronze_layer b
    JOIN silver_layer s
        ON b.id = s.id
    WHERE
        COALESCE(CAST(b.processid AS VARCHAR), '') <> COALESCE(CAST(s.processid AS VARCHAR), '')
        OR COALESCE(CAST(b.tickettypeid AS VARCHAR), '') <> COALESCE(CAST(s.tickettype AS VARCHAR), '')
        OR COALESCE(CAST(b.ticketstatusid AS VARCHAR), '') <> COALESCE(CAST(s.ticketstatus AS VARCHAR), '')
        OR COALESCE(CAST(b.entityidentifier AS VARCHAR), '') <> COALESCE(CAST(s.entityidentifier AS VARCHAR), '')
        OR COALESCE(CAST(b.refnumber AS VARCHAR), '') <> COALESCE(CAST(s.refnumber AS VARCHAR), '')
        OR COALESCE(CAST(b.closedon AS VARCHAR), '') <> COALESCE(CAST(s.closedon AS VARCHAR), '')
        OR COALESCE(CAST(b.isactive AS VARCHAR), '') <> COALESCE(CAST(s.isactive AS VARCHAR), '')
        OR COALESCE(CAST(b.insertedpassword AS VARCHAR), '') <> COALESCE(CAST(s.insertedpassword AS VARCHAR), '')
        OR COALESCE(CAST(b.tkchannelid AS VARCHAR), '') <> COALESCE(CAST(s.tkchannel AS VARCHAR), '')
) t;
