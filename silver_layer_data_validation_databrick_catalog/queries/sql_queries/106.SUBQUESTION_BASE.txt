WITH bronze_layer AS (
    SELECT
        id,
        questionid,
        subquestion,
        subquestionar,
        answertypeid,
        lookuptypeid,
        lookupmultiseltypeid,
        dependantsubquestionid,
        dependantsubquestionanswer,
        dependantsubquestionoperator,
        dependantreversevalue,
        externalid,
        autocalc_value1,
        autocalc_operator,
        autocalc_value2,
        "order",
        listcol,
        ismandatory,
        isfullwidth,
        FALSE AS is_deleted
    FROM dev_iceberg."tmkn-aws-dwh-dev-iceberg-bronze".OSUSR_5W2_SUBQUESTION
),

silver_layer AS (
    SELECT
        id,
        questionid,
        subquestion,
        subquestionar,
        answertypeid,
        lookuptypeid,
        lookupmultiseltypeid,
        dependantsubquestionid,
        dependantsubquestionanswer,
        dependantsubquestionoperator,
        dependantreversevalue,
        externalid,
        autocalc_value1,
        autocalc_operator,
        autocalc_value2,
        "order",
        listcol,
        ismandatory,
        isfullwidth,
        FALSE AS is_deleted
    FROM dev_iceberg."tmkn-aws-dwh-dev-iceberg-silver".SUBQUESTION_BASE
)

-- =========================
-- VALIDATION
-- =========================

SELECT 'COUNT_VALIDATION', COUNT(*), (SELECT COUNT(*) FROM silver_layer)
FROM bronze_layer

UNION ALL

SELECT 'DUPLICATE_BRONZE', COUNT(*), NULL
FROM (
    SELECT id
    FROM bronze_layer
    GROUP BY id
    HAVING COUNT(*) > 1
) t

UNION ALL

SELECT 'DUPLICATE_SILVER', COUNT(*), NULL
FROM (
    SELECT id
    FROM silver_layer
    GROUP BY id
    HAVING COUNT(*) > 1
) t

UNION ALL

SELECT 'PK_NULL_BRONZE', COUNT(*), NULL
FROM bronze_layer
WHERE id IS NULL

UNION ALL

SELECT 'PK_NULL_SILVER', COUNT(*), NULL
FROM silver_layer
WHERE id IS NULL

UNION ALL

SELECT 'BRONZE_NOT_IN_SILVER', COUNT(*), NULL
FROM (
    SELECT
        CAST(id AS VARCHAR),
        CAST(subquestion AS VARCHAR),
        CAST(questionid AS VARCHAR)
    FROM bronze_layer

    EXCEPT

    SELECT
        CAST(id AS VARCHAR),
        CAST(subquestion AS VARCHAR),
        CAST(questionid AS VARCHAR)
    FROM silver_layer
) t

UNION ALL

SELECT 'SILVER_NOT_IN_BRONZE', COUNT(*), NULL
FROM (
    SELECT
        CAST(id AS VARCHAR),
        CAST(subquestion AS VARCHAR),
        CAST(questionid AS VARCHAR)
    FROM silver_layer

    EXCEPT

    SELECT
        CAST(id AS VARCHAR),
        CAST(subquestion AS VARCHAR),
        CAST(questionid AS VARCHAR)
    FROM bronze_layer
) t

UNION ALL

SELECT 'COLUMN_MISMATCH_COUNT', COUNT(*), NULL
FROM (
    SELECT b.id
    FROM bronze_layer b
    JOIN silver_layer s
        ON b.id = s.id
    WHERE
        COALESCE(CAST(b.questionid AS VARCHAR), '') <> COALESCE(CAST(s.questionid AS VARCHAR), '')
        OR COALESCE(CAST(b.subquestion AS VARCHAR), '') <> COALESCE(CAST(s.subquestion AS VARCHAR), '')
        OR COALESCE(CAST(b.subquestionar AS VARCHAR), '') <> COALESCE(CAST(s.subquestionar AS VARCHAR), '')
        OR COALESCE(CAST(b.answertypeid AS VARCHAR), '') <> COALESCE(CAST(s.answertypeid AS VARCHAR), '')
        OR COALESCE(CAST(b.lookuptypeid AS VARCHAR), '') <> COALESCE(CAST(s.lookuptypeid AS VARCHAR), '')
        OR COALESCE(CAST(b.lookupmultiseltypeid AS VARCHAR), '') <> COALESCE(CAST(s.lookupmultiseltypeid AS VARCHAR), '')
        OR COALESCE(CAST(b.dependantsubquestionid AS VARCHAR), '') <> COALESCE(CAST(s.dependantsubquestionid AS VARCHAR), '')
        OR COALESCE(CAST(b.dependantsubquestionanswer AS VARCHAR), '') <> COALESCE(CAST(s.dependantsubquestionanswer AS VARCHAR), '')
        OR COALESCE(CAST(b.dependantsubquestionoperator AS VARCHAR), '') <> COALESCE(CAST(s.dependantsubquestionoperator AS VARCHAR), '')
        OR COALESCE(CAST(b.dependantreversevalue AS VARCHAR), '') <> COALESCE(CAST(s.dependantreversevalue AS VARCHAR), '')
        OR COALESCE(CAST(b.externalid AS VARCHAR), '') <> COALESCE(CAST(s.externalid AS VARCHAR), '')
        OR COALESCE(CAST(b.autocalc_value1 AS VARCHAR), '') <> COALESCE(CAST(s.autocalc_value1 AS VARCHAR), '')
        OR COALESCE(CAST(b.autocalc_operator AS VARCHAR), '') <> COALESCE(CAST(s.autocalc_operator AS VARCHAR), '')
        OR COALESCE(CAST(b.autocalc_value2 AS VARCHAR), '') <> COALESCE(CAST(s.autocalc_value2 AS VARCHAR), '')
        OR COALESCE(CAST(b."order" AS VARCHAR), '') <> COALESCE(CAST(s."order" AS VARCHAR), '')
        OR COALESCE(CAST(b.listcol AS VARCHAR), '') <> COALESCE(CAST(s.listcol AS VARCHAR), '')
        OR COALESCE(CAST(b.ismandatory AS VARCHAR), '') <> COALESCE(CAST(s.ismandatory AS VARCHAR), '')
        OR COALESCE(CAST(b.isfullwidth AS VARCHAR), '') <> COALESCE(CAST(s.isfullwidth AS VARCHAR), '')
) t;
