WITH bronze_layer AS (
    SELECT
        id,
        specialconditionid,
        processid,
        remarkscustomer,
        remarkstk,
        submissiondate,
        referencenumber,
        isactive,
        createdon,
        updatedon,
        submittedby,
        tkremarkupdatedby,
        tkremarkupdatedon,
        conditiontype
    FROM dev_iceberg."tmkn-aws-dwh-dev-iceberg-bronze".OSUSR_765_SPECIALCONDITIONFULFILMENT
),

silver_layer AS (
    SELECT
        id,
        specialconditionid,
        processid,
        remarkscustomer,
        remarkstk,
        submissiondate,
        referencenumber,
        is_deleted AS isactive,
        createdon,
        updatedon,
        submittedby,
        tkremarkupdatedby,
        tkremarkupdatedon,
        conditiontype
    FROM dev_iceberg."tmkn-aws-dwh-dev-iceberg-silver".SPECIAL_CONDITION_FULFILMENT_BASE
)

-- =========================
-- VALIDATION
-- =========================

-- COUNT VALIDATION
SELECT 
    'COUNT_VALIDATION' AS validation_type,
    (SELECT COUNT(*) FROM bronze_layer) AS source_count,
    (SELECT COUNT(*) FROM silver_layer) AS target_count,
    (SELECT COUNT(*) FROM bronze_layer) - (SELECT COUNT(*) FROM silver_layer) AS difference

UNION ALL

-- DUPLICATE BRONZE
SELECT 
    'DUPLICATE_BRONZE',
    COUNT(*),
    NULL,
    NULL
FROM (
    SELECT id
    FROM bronze_layer
    GROUP BY id
    HAVING COUNT(*) > 1
)

UNION ALL

-- DUPLICATE SILVER
SELECT 
    'DUPLICATE_SILVER',
    COUNT(*),
    NULL,
    NULL
FROM (
    SELECT id
    FROM silver_layer
    GROUP BY id
    HAVING COUNT(*) > 1
)

UNION ALL

-- PK NULL BRONZE
SELECT 
    'PK_NULL_BRONZE',
    COUNT(*),
    NULL,
    NULL
FROM bronze_layer
WHERE id IS NULL

UNION ALL

-- PK NULL SILVER
SELECT 
    'PK_NULL_SILVER',
    COUNT(*),
    NULL,
    NULL
FROM silver_layer
WHERE id IS NULL

UNION ALL

-- BRONZE NOT IN SILVER
SELECT 
    'BRONZE_NOT_IN_SILVER',
    COUNT(*),
    NULL,
    NULL
FROM (
    SELECT 
        CAST(id AS VARCHAR),
        CAST(specialconditionid AS VARCHAR),
        CAST(processid AS VARCHAR)
    FROM bronze_layer

    EXCEPT

    SELECT 
        CAST(id AS VARCHAR),
        CAST(specialconditionid AS VARCHAR),
        CAST(processid AS VARCHAR)
    FROM silver_layer
)

UNION ALL

-- SILVER NOT IN BRONZE
SELECT 
    'SILVER_NOT_IN_BRONZE',
    COUNT(*),
    NULL,
    NULL
FROM (
    SELECT 
        CAST(id AS VARCHAR),
        CAST(specialconditionid AS VARCHAR),
        CAST(processid AS VARCHAR)
    FROM silver_layer

    EXCEPT

    SELECT 
        CAST(id AS VARCHAR),
        CAST(specialconditionid AS VARCHAR),
        CAST(processid AS VARCHAR)
    FROM bronze_layer
)

UNION ALL

-- COLUMN MISMATCH CHECK
SELECT 
    'COLUMN_MISMATCH_COUNT',
    COUNT(*),
    NULL,
    NULL
FROM (
    SELECT *
    FROM bronze_layer b
    JOIN silver_layer s
        ON b.id = s.id
    WHERE 
        COALESCE(CAST(b.specialconditionid AS VARCHAR),'') 
        IS DISTINCT FROM COALESCE(CAST(s.specialconditionid AS VARCHAR),'')
     OR COALESCE(CAST(b.processid AS VARCHAR),'') 
        IS DISTINCT FROM COALESCE(CAST(s.processid AS VARCHAR),'')
     OR COALESCE(CAST(b.remarkscustomer AS VARCHAR),'') 
        IS DISTINCT FROM COALESCE(CAST(s.remarkscustomer AS VARCHAR),'')
     OR COALESCE(CAST(b.remarkstk AS VARCHAR),'') 
        IS DISTINCT FROM COALESCE(CAST(s.remarkstk AS VARCHAR),'')
     OR COALESCE(CAST(b.submissiondate AS VARCHAR),'') 
        IS DISTINCT FROM COALESCE(CAST(s.submissiondate AS VARCHAR),'')
     OR COALESCE(CAST(b.referencenumber AS VARCHAR),'') 
        IS DISTINCT FROM COALESCE(CAST(s.referencenumber AS VARCHAR),'')
     OR COALESCE(CAST(b.isactive AS VARCHAR),'') 
        IS DISTINCT FROM COALESCE(CAST(s.isactive AS VARCHAR),'')
     OR COALESCE(CAST(b.createdon AS VARCHAR),'') 
        IS DISTINCT FROM COALESCE(CAST(s.createdon AS VARCHAR),'')
     OR COALESCE(CAST(b.updatedon AS VARCHAR),'') 
        IS DISTINCT FROM COALESCE(CAST(s.updatedon AS VARCHAR),'')
     OR COALESCE(CAST(b.submittedby AS VARCHAR),'') 
        IS DISTINCT FROM COALESCE(CAST(s.submittedby AS VARCHAR),'')
     OR COALESCE(CAST(b.tkremarkupdatedby AS VARCHAR),'') 
        IS DISTINCT FROM COALESCE(CAST(s.tkremarkupdatedby AS VARCHAR),'')
     OR COALESCE(CAST(b.tkremarkupdatedon AS VARCHAR),'') 
        IS DISTINCT FROM COALESCE(CAST(s.tkremarkupdatedon AS VARCHAR),'')
     OR COALESCE(CAST(b.conditiontype AS VARCHAR),'') 
        IS DISTINCT FROM COALESCE(CAST(s.conditiontype AS VARCHAR),'')
) t;
