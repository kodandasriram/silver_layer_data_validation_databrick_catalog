---78) dev_iceberg."tmkn-aws-dwh-dev-iceberg-silver".SUPPORT_STRUCTURE_BASE


WITH bronze_layer AS (
    SELECT
        applicationsupportid,
        programsupporttypesubitemid,
        description,
        requestedamt,
        tkshareovr,
        tkshare,
        iseligible,
        eligibleerrormesage,
        duedate,
        requiresinspection,
        itemamtavailable,
        itemamtinprogress,
        itemamtclaimed,
        numberofclaims,
        createdon,
        updatedon,
        FALSE AS is_deleted
    FROM dev_iceberg."tmkn-aws-dwh-dev-iceberg-bronze".OSUSR_2DA_SUPPORTSTRUCTURE
),

silver_layer AS (
    SELECT
        applicationsupportid,
        programsupporttypesubitemid,
        description,
        requestedamt,
        tkshareovr,
        tkshare,
        iseligible,
        eligibleerrormesage,
        duedate,
        requiresinspection,
        itemamtavailable,
        itemamtinprogress,
        itemamtclaimed,
        numberofclaims,
        createdon,
        updatedon,
        is_deleted
    FROM dev_iceberg."tmkn-aws-dwh-dev-iceberg-silver".SUPPORT_STRUCTURE_BASE
)

-- =========================
-- VALIDATION
-- =========================

SELECT
    'COUNT_VALIDATION' AS validation_type,
    COUNT(*) AS bronze_count,
    (SELECT COUNT(*) FROM silver_layer) AS silver_count
FROM bronze_layer

UNION ALL

SELECT
    'DUPLICATE_BRONZE',
    COUNT(*),
    NULL
FROM (
    SELECT applicationsupportid
    FROM bronze_layer
    GROUP BY applicationsupportid
    HAVING COUNT(*) > 1
)

UNION ALL

SELECT
    'DUPLICATE_SILVER',
    COUNT(*),
    NULL
FROM (
    SELECT applicationsupportid
    FROM silver_layer
    GROUP BY applicationsupportid
    HAVING COUNT(*) > 1
)

UNION ALL

SELECT
    'PK_NULL_BRONZE',
    COUNT(*),
    NULL
FROM bronze_layer
WHERE applicationsupportid IS NULL

UNION ALL

SELECT
    'PK_NULL_SILVER',
    COUNT(*),
    NULL
FROM silver_layer
WHERE applicationsupportid IS NULL

UNION ALL

SELECT
    'BRONZE_NOT_IN_SILVER',
    COUNT(*),
    NULL
FROM (
    SELECT
        CAST(applicationsupportid AS VARCHAR),
        CAST(description AS VARCHAR),
        CAST(requestedamt AS VARCHAR)
    FROM bronze_layer

    EXCEPT

    SELECT
        CAST(applicationsupportid AS VARCHAR),
        CAST(description AS VARCHAR),
        CAST(requestedamt AS VARCHAR)
    FROM silver_layer
)

UNION ALL

SELECT
    'SILVER_NOT_IN_BRONZE',
    COUNT(*),
    NULL
FROM (
    SELECT
        CAST(applicationsupportid AS VARCHAR),
        CAST(description AS VARCHAR),
        CAST(requestedamt AS VARCHAR)
    FROM silver_layer

    EXCEPT

    SELECT
        CAST(applicationsupportid AS VARCHAR),
        CAST(description AS VARCHAR),
        CAST(requestedamt AS VARCHAR)
    FROM bronze_layer
)

UNION ALL

SELECT
    'COLUMN_MISMATCH_COUNT',
    COUNT(*),
    NULL
FROM (
    SELECT *
    FROM bronze_layer b
    JOIN silver_layer s
        ON b.applicationsupportid = s.applicationsupportid
    WHERE
           COALESCE(CAST(b.programsupporttypesubitemid AS VARCHAR), '') <> COALESCE(CAST(s.programsupporttypesubitemid AS VARCHAR), '')
        OR COALESCE(CAST(b.description AS VARCHAR), '') <> COALESCE(CAST(s.description AS VARCHAR), '')
        OR COALESCE(CAST(b.requestedamt AS VARCHAR), '') <> COALESCE(CAST(s.requestedamt AS VARCHAR), '')
        OR COALESCE(CAST(b.tkshareovr AS VARCHAR), '') <> COALESCE(CAST(s.tkshareovr AS VARCHAR), '')
        OR COALESCE(CAST(b.tkshare AS VARCHAR), '') <> COALESCE(CAST(s.tkshare AS VARCHAR), '')
        OR COALESCE(CAST(b.iseligible AS VARCHAR), '') <> COALESCE(CAST(s.iseligible AS VARCHAR), '')
        OR COALESCE(CAST(b.eligibleerrormesage AS VARCHAR), '') <> COALESCE(CAST(s.eligibleerrormesage AS VARCHAR), '')
        OR COALESCE(CAST(b.duedate AS VARCHAR), '') <> COALESCE(CAST(s.duedate AS VARCHAR), '')
        OR COALESCE(CAST(b.requiresinspection AS VARCHAR), '') <> COALESCE(CAST(s.requiresinspection AS VARCHAR), '')
        OR COALESCE(CAST(b.itemamtavailable AS VARCHAR), '') <> COALESCE(CAST(s.itemamtavailable AS VARCHAR), '')
        OR COALESCE(CAST(b.itemamtinprogress AS VARCHAR), '') <> COALESCE(CAST(s.itemamtinprogress AS VARCHAR), '')
        OR COALESCE(CAST(b.itemamtclaimed AS VARCHAR), '') <> COALESCE(CAST(s.itemamtclaimed AS VARCHAR), '')
        OR COALESCE(CAST(b.numberofclaims AS VARCHAR), '') <> COALESCE(CAST(s.numberofclaims AS VARCHAR), '')
        OR COALESCE(CAST(b.createdon AS VARCHAR), '') <> COALESCE(CAST(s.createdon AS VARCHAR), '')
        OR COALESCE(CAST(b.updatedon AS VARCHAR), '') <> COALESCE(CAST(s.updatedon AS VARCHAR), '')
) t;
