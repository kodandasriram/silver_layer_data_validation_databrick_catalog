WITH bronze_layer AS (
    SELECT
        id,
        programversionid,
        supportareaid,
        eligibilityrequesttypeid,
        eligibilitycriteriatypeid,
        name,
        description,
        expression,
        createdby,
        createdon,
        updatedby,
        updatedon,
        errormessage_en,
        errormessage_ar,
        programsupporttypesubitemid,
        eligibilitypreventiontypeid,
        supportstipendextrashare,
        supporttypeid,
        supportedemployeetype
    FROM dev_iceberg."tmkn-aws-dwh-dev-iceberg-bronze".OSUSR_3QQ_SUPPORTAREAELIGIBILITYCRITERIA
),

silver_layer AS (
    SELECT
        id,
        programversionid,
        supportarea AS supportareaid,
        eligibilityrequesttype AS eligibilityrequesttypeid,
        eligibilitycriteriatype AS eligibilitycriteriatypeid,
        name,
        description,
        expression,
        createdby,
        createdon,
        updatedby,
        updatedon,
        errormessage_en,
        errormessage_ar,
        programsupporttypesubitem AS programsupporttypesubitemid,
        eligibilitypreventiontype AS eligibilitypreventiontypeid,
        supportstipendextrashare,
        supporttype AS supporttypeid,
        supportedemployeetype
    FROM dev_iceberg."tmkn-aws-dwh-dev-iceberg-silver".support_eligibility_criteria_base
)

-- =========================
-- VALIDATION
-- =========================

-- COUNT VALIDATION
SELECT
    'COUNT_VALIDATION' AS validation_type,
    COUNT(*) AS bronze_count,
    (SELECT COUNT(*) FROM silver_layer) AS silver_count
FROM bronze_layer

UNION ALL

-- DUPLICATE CHECK - BRONZE
SELECT
    'DUPLICATE_BRONZE',
    COUNT(*),
    NULL
FROM (
    SELECT id
    FROM bronze_layer
    GROUP BY id
    HAVING COUNT(*) > 1
)

UNION ALL

-- DUPLICATE CHECK - SILVER
SELECT
    'DUPLICATE_SILVER',
    COUNT(*),
    NULL
FROM (
    SELECT id
    FROM silver_layer
    GROUP BY id
    HAVING COUNT(*) > 1
)

UNION ALL

-- PRIMARY KEY NULL CHECK - BRONZE
SELECT
    'PK_NULL_BRONZE',
    COUNT(*),
    NULL
FROM bronze_layer
WHERE id IS NULL

UNION ALL

-- PRIMARY KEY NULL CHECK - SILVER
SELECT
    'PK_NULL_SILVER',
    COUNT(*),
    NULL
FROM silver_layer
WHERE id IS NULL

UNION ALL

-- BRONZE NOT IN SILVER
SELECT
    'BRONZE_NOT_IN_SILVER',
    COUNT(*),
    NULL
FROM (
    SELECT
        CAST(id AS VARCHAR),
        CAST(name AS VARCHAR),
        CAST(programversionid AS VARCHAR)
    FROM bronze_layer

    EXCEPT

    SELECT
        CAST(id AS VARCHAR),
        CAST(name AS VARCHAR),
        CAST(programversionid AS VARCHAR)
    FROM silver_layer
)

UNION ALL

-- SILVER NOT IN BRONZE
SELECT
    'SILVER_NOT_IN_BRONZE',
    COUNT(*),
    NULL
FROM (
    SELECT
        CAST(id AS VARCHAR),
        CAST(name AS VARCHAR),
        CAST(programversionid AS VARCHAR)
    FROM silver_layer

    EXCEPT

    SELECT
        CAST(id AS VARCHAR),
        CAST(name AS VARCHAR),
        CAST(programversionid AS VARCHAR)
    FROM bronze_layer
)

UNION ALL

-- COLUMN MISMATCH CHECK
SELECT
    'COLUMN_MISMATCH_COUNT',
    COUNT(*),
    NULL
FROM (
    SELECT *
    FROM bronze_layer b
    JOIN silver_layer s
        ON b.id = s.id
    WHERE
           COALESCE(CAST(b.programversionid AS VARCHAR), '') <> COALESCE(CAST(s.programversionid AS VARCHAR), '')
        OR COALESCE(CAST(b.supportareaid AS VARCHAR), '') <> COALESCE(CAST(s.supportareaid AS VARCHAR), '')
        OR COALESCE(CAST(b.eligibilityrequesttypeid AS VARCHAR), '') <> COALESCE(CAST(s.eligibilityrequesttypeid AS VARCHAR), '')
        OR COALESCE(CAST(b.eligibilitycriteriatypeid AS VARCHAR), '') <> COALESCE(CAST(s.eligibilitycriteriatypeid AS VARCHAR), '')
        OR COALESCE(TRIM(CAST(b.name AS VARCHAR)), '') <> COALESCE(TRIM(CAST(s.name AS VARCHAR)), '')
        OR COALESCE(TRIM(CAST(b.description AS VARCHAR)), '') <> COALESCE(TRIM(CAST(s.description AS VARCHAR)), '')
        OR COALESCE(TRIM(CAST(b.expression AS VARCHAR)), '') <> COALESCE(TRIM(CAST(s.expression AS VARCHAR)), '')
        OR COALESCE(CAST(b.createdby AS VARCHAR), '') <> COALESCE(CAST(s.createdby AS VARCHAR), '')
        OR COALESCE(CAST(b.createdon AS VARCHAR), '') <> COALESCE(CAST(s.createdon AS VARCHAR), '')
        OR COALESCE(CAST(b.updatedby AS VARCHAR), '') <> COALESCE(CAST(s.updatedby AS VARCHAR), '')
        OR COALESCE(CAST(b.updatedon AS VARCHAR), '') <> COALESCE(CAST(s.updatedon AS VARCHAR), '')
        OR COALESCE(TRIM(CAST(b.errormessage_en AS VARCHAR)), '') <> COALESCE(TRIM(CAST(s.errormessage_en AS VARCHAR)), '')
        OR COALESCE(TRIM(CAST(b.errormessage_ar AS VARCHAR)), '') <> COALESCE(TRIM(CAST(s.errormessage_ar AS VARCHAR)), '')
        OR COALESCE(CAST(b.programsupporttypesubitemid AS VARCHAR), '') <> COALESCE(CAST(s.programsupporttypesubitemid AS VARCHAR), '')
        OR COALESCE(CAST(b.eligibilitypreventiontypeid AS VARCHAR), '') <> COALESCE(CAST(s.eligibilitypreventiontypeid AS VARCHAR), '')
        OR COALESCE(CAST(b.supportstipendextrashare AS VARCHAR), '') <> COALESCE(CAST(s.supportstipendextrashare AS VARCHAR), '')
        OR COALESCE(CAST(b.supporttypeid AS VARCHAR), '') <> COALESCE(CAST(s.supporttypeid AS VARCHAR), '')
        OR COALESCE(TRIM(CAST(b.supportedemployeetype AS VARCHAR)), '') <> COALESCE(TRIM(CAST(s.supportedemployeetype AS VARCHAR)), '')
) t;
