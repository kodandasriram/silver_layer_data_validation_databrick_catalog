WITH bronze_layer AS (
    SELECT
        tmkn_violationid,
        tmkn_violationno,
        tmkn_crlicenseno,
        tmkn_sendwarningletter,
        tmkn_bcviolation,
        tmkn_payorganizer,
        tmkn_sendtoeconomiccrimesdivision,
        tmkn_sendforsitevisitinspection,
        tmkn_sendtocentralinformaticsorganisation,
        tmkn_forcemajeure,
        tmkn_addtoblacklist,
        tmkn_otherdecisions,
        tmkn_batchcreated,
        tmkn_supportbc,
        tmkn_encashguaranteecheque,
        tmkn_bcwithdrawal,
        tmkn_other,
        tmkn_cancellationofexhibition,
        FALSE AS is_deleted
    FROM dev_iceberg."tmkn-aws-dwh-dev-iceberg-bronze".TMKN_VIOLATIONBASE
),

silver_layer AS (
    SELECT
        tmkn_violationid,
        tmkn_violationno,
        tmkn_crlicenseno,
        tmkn_sendwarningletter,
        tmkn_bcviolation,
        tmkn_payorganizer,
        tmkn_sendtoeconomiccrimesdivision,
        tmkn_sendforsitevisitinspection,
        tmkn_sendtocentralinformaticsorganisation,
        tmkn_forcemajeure,
        tmkn_addtoblacklist,
        tmkn_otherdecisions,
        tmkn_batchcreated,
        tmkn_supportbc,
        tmkn_encashguaranteecheque,
        tmkn_bcwithdrawal,
        tmkn_other,
        tmkn_cancellationofexhibition,
        FALSE AS is_deleted
    FROM dev_iceberg."tmkn-aws-dwh-dev-iceberg-silver".SLV_MIS_VIOLATION
)

-- =========================
-- VALIDATION
-- =========================

SELECT 'COUNT_VALIDATION', COUNT(*), (SELECT COUNT(*) FROM silver_layer)
FROM bronze_layer

UNION ALL

SELECT 'DUPLICATE_BRONZE', COUNT(*), NULL
FROM (
    SELECT tmkn_violationid
    FROM bronze_layer
    GROUP BY tmkn_violationid
    HAVING COUNT(*) > 1
) t

UNION ALL

SELECT 'DUPLICATE_SILVER', COUNT(*), NULL
FROM (
    SELECT tmkn_violationid
    FROM silver_layer
    GROUP BY tmkn_violationid
    HAVING COUNT(*) > 1
) t

UNION ALL

SELECT 'PK_NULL_BRONZE', COUNT(*), NULL
FROM bronze_layer
WHERE tmkn_violationid IS NULL

UNION ALL

SELECT 'PK_NULL_SILVER', COUNT(*), NULL
FROM silver_layer
WHERE tmkn_violationid IS NULL

UNION ALL

SELECT 'BRONZE_NOT_IN_SILVER', COUNT(*), NULL
FROM (
    SELECT
        CAST(tmkn_violationid AS VARCHAR),
        CAST(tmkn_violationno AS VARCHAR),
        CAST(tmkn_crlicenseno AS VARCHAR)
    FROM bronze_layer

    EXCEPT

    SELECT
        CAST(tmkn_violationid AS VARCHAR),
        CAST(tmkn_violationno AS VARCHAR),
        CAST(tmkn_crlicenseno AS VARCHAR)
    FROM silver_layer
) t

UNION ALL

SELECT 'SILVER_NOT_IN_BRONZE', COUNT(*), NULL
FROM (
    SELECT
        CAST(tmkn_violationid AS VARCHAR),
        CAST(tmkn_violationno AS VARCHAR),
        CAST(tmkn_crlicenseno AS VARCHAR)
    FROM silver_layer

    EXCEPT

    SELECT
        CAST(tmkn_violationid AS VARCHAR),
        CAST(tmkn_violationno AS VARCHAR),
        CAST(tmkn_crlicenseno AS VARCHAR)
    FROM bronze_layer
) t

UNION ALL

SELECT 'COLUMN_MISMATCH_COUNT', COUNT(*), NULL
FROM (
    SELECT b.tmkn_violationid
    FROM bronze_layer b
    JOIN silver_layer s
        ON b.tmkn_violationid = s.tmkn_violationid
    WHERE
        COALESCE(CAST(b.tmkn_violationno AS VARCHAR), '') <> COALESCE(CAST(s.tmkn_violationno AS VARCHAR), '')
        OR COALESCE(CAST(b.tmkn_crlicenseno AS VARCHAR), '') <> COALESCE(CAST(s.tmkn_crlicenseno AS VARCHAR), '')
        OR COALESCE(CAST(b.tmkn_sendwarningletter AS VARCHAR), '') <> COALESCE(CAST(s.tmkn_sendwarningletter AS VARCHAR), '')
        OR COALESCE(CAST(b.tmkn_bcviolation AS VARCHAR), '') <> COALESCE(CAST(s.tmkn_bcviolation AS VARCHAR), '')
        OR COALESCE(CAST(b.tmkn_payorganizer AS VARCHAR), '') <> COALESCE(CAST(s.tmkn_payorganizer AS VARCHAR), '')
        OR COALESCE(CAST(b.tmkn_sendtoeconomiccrimesdivision AS VARCHAR), '') <> COALESCE(CAST(s.tmkn_sendtoeconomiccrimesdivision AS VARCHAR), '')
        OR COALESCE(CAST(b.tmkn_sendforsitevisitinspection AS VARCHAR), '') <> COALESCE(CAST(s.tmkn_sendforsitevisitinspection AS VARCHAR), '')
        OR COALESCE(CAST(b.tmkn_sendtocentralinformaticsorganisation AS VARCHAR), '') <> COALESCE(CAST(s.tmkn_sendtocentralinformaticsorganisation AS VARCHAR), '')
        OR COALESCE(CAST(b.tmkn_forcemajeure AS VARCHAR), '') <> COALESCE(CAST(s.tmkn_forcemajeure AS VARCHAR), '')
        OR COALESCE(CAST(b.tmkn_addtoblacklist AS VARCHAR), '') <> COALESCE(CAST(s.tmkn_addtoblacklist AS VARCHAR), '')
        OR COALESCE(CAST(b.tmkn_otherdecisions AS VARCHAR), '') <> COALESCE(CAST(s.tmkn_otherdecisions AS VARCHAR), '')
        OR COALESCE(CAST(b.tmkn_batchcreated AS VARCHAR), '') <> COALESCE(CAST(s.tmkn_batchcreated AS VARCHAR), '')
        OR COALESCE(CAST(b.tmkn_supportbc AS VARCHAR), '') <> COALESCE(CAST(s.tmkn_supportbc AS VARCHAR), '')
        OR COALESCE(CAST(b.tmkn_encashguaranteecheque AS VARCHAR), '') <> COALESCE(CAST(s.tmkn_encashguaranteecheque AS VARCHAR), '')
        OR COALESCE(CAST(b.tmkn_bcwithdrawal AS VARCHAR), '') <> COALESCE(CAST(s.tmkn_bcwithdrawal AS VARCHAR), '')
        OR COALESCE(CAST(b.tmkn_other AS VARCHAR), '') <> COALESCE(CAST(s.tmkn_other AS VARCHAR), '')
        OR COALESCE(CAST(b.tmkn_cancellationofexhibition AS VARCHAR), '') <> COALESCE(CAST(s.tmkn_cancellationofexhibition AS VARCHAR), '')
) t;
