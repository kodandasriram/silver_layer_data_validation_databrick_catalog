---92) dev_iceberg."tmkn-aws-dwh-dev-iceberg-silver".ZMZ_CUSTOMER_BASE

WITH bronze_layer AS (
SELECT
id,
createdon,
updatedon,
FALSE AS is_deleted
FROM dev_iceberg."tmkn-aws-dwh-dev-iceberg-bronze".OSUSR_ZMZ_CUSTOMER
),

silver_layer AS (
SELECT
id,
createdon,
updatedon
FROM dev_iceberg."tmkn-aws-dwh-dev-iceberg-silver".ZMZ_CUSTOMER_BASE
)

-- =========================
-- VALIDATION
-- =========================

-- COUNT
SELECT 'COUNT_VALIDATION' AS validation_type,
COUNT(*) AS bronze_count,
(SELECT COUNT(*) FROM silver_layer) AS silver_count
FROM bronze_layer

UNION ALL

-- DUPLICATES
SELECT 'DUPLICATE_BRONZE', COUNT(*), NULL
FROM (
SELECT id
FROM bronze_layer
GROUP BY id
HAVING COUNT(*) > 1
) b

UNION ALL

SELECT 'DUPLICATE_SILVER', COUNT(*), NULL
FROM (
SELECT id
FROM silver_layer
GROUP BY id
HAVING COUNT(*) > 1
) s

UNION ALL

-- PK NULL CHECK
SELECT 'PK_NULL_BRONZE', COUNT(*), NULL
FROM bronze_layer
WHERE id IS NULL

UNION ALL

SELECT 'PK_NULL_SILVER', COUNT(*), NULL
FROM silver_layer
WHERE id IS NULL

UNION ALL

-- BRONZE NOT IN SILVER
SELECT 'BRONZE_NOT_IN_SILVER', COUNT(*), NULL
FROM (
SELECT
CAST(id AS VARCHAR),
CAST(createdon AS VARCHAR),
CAST(updatedon AS VARCHAR)
FROM bronze_layer


EXCEPT

SELECT
    CAST(id AS VARCHAR),
    CAST(createdon AS VARCHAR),
    CAST(updatedon AS VARCHAR)
FROM silver_layer


) bns

UNION ALL

-- SILVER NOT IN BRONZE
SELECT 'SILVER_NOT_IN_BRONZE', COUNT(*), NULL
FROM (
SELECT
CAST(id AS VARCHAR),
CAST(createdon AS VARCHAR),
CAST(updatedon AS VARCHAR)
FROM silver_layer


EXCEPT

SELECT
    CAST(id AS VARCHAR),
    CAST(createdon AS VARCHAR),
    CAST(updatedon AS VARCHAR)
FROM bronze_layer


) snb

UNION ALL

-- COLUMN MISMATCH COUNT
SELECT 'COLUMN_MISMATCH_COUNT', COUNT(*), NULL
FROM (
SELECT b.id
FROM bronze_layer b
JOIN silver_layer s
ON CAST(b.id AS VARCHAR) = CAST(s.id AS VARCHAR)
WHERE COALESCE(CAST(b.createdon AS VARCHAR), '') <> COALESCE(CAST(s.createdon AS VARCHAR), '')
OR COALESCE(CAST(b.updatedon AS VARCHAR), '') <> COALESCE(CAST(s.updatedon AS VARCHAR), '')
) t;
