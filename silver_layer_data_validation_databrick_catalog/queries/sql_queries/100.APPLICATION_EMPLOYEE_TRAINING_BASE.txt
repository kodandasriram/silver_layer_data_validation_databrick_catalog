-- =========================
-- VALIDATION
-- =========================

-- COUNT
SELECT 'COUNT_VALIDATION', COUNT(*), (SELECT COUNT(*) FROM silver_layer)
FROM bronze_layer

UNION ALL

-- DUPLICATES BRONZE
SELECT 'DUPLICATE_BRONZE', COUNT(*), NULL
FROM (
    SELECT applicationemployeeid
    FROM bronze_layer
    GROUP BY applicationemployeeid
    HAVING COUNT(*) > 1
)

UNION ALL

-- DUPLICATES SILVER
SELECT 'DUPLICATE_SILVER', COUNT(*), NULL
FROM (
    SELECT applicationemployeeid
    FROM silver_layer
    GROUP BY applicationemployeeid
    HAVING COUNT(*) > 1
)

UNION ALL

-- PK NULL BRONZE
SELECT 'PK_NULL_BRONZE', COUNT(*), NULL
FROM bronze_layer
WHERE applicationemployeeid IS NULL

UNION ALL

-- PK NULL SILVER
SELECT 'PK_NULL_SILVER', COUNT(*), NULL
FROM silver_layer
WHERE applicationemployeeid IS NULL

UNION ALL

-- BRONZE NOT IN SILVER
SELECT 'BRONZE_NOT_IN_SILVER', COUNT(*), NULL
FROM (
    SELECT 
        CAST(applicationemployeeid AS VARCHAR),
        CAST(nameofcoursecertification AS VARCHAR),
        CAST(costoftraining AS VARCHAR)
    FROM bronze_layer

    EXCEPT

    SELECT 
        CAST(applicationemployeeid AS VARCHAR),
        CAST(nameofcoursecertification AS VARCHAR),
        CAST(costoftraining AS VARCHAR)
    FROM silver_layer
)

UNION ALL

-- SILVER NOT IN BRONZE
SELECT 'SILVER_NOT_IN_BRONZE', COUNT(*), NULL
FROM (
    SELECT 
        CAST(applicationemployeeid AS VARCHAR),
        CAST(nameofcoursecertification AS VARCHAR),
        CAST(costoftraining AS VARCHAR)
    FROM silver_layer

    EXCEPT

    SELECT 
        CAST(applicationemployeeid AS VARCHAR),
        CAST(nameofcoursecertification AS VARCHAR),
        CAST(costoftraining AS VARCHAR)
    FROM bronze_layer
)

UNION ALL

-- COLUMN MISMATCH COUNT
SELECT 'COLUMN_MISMATCH_COUNT', COUNT(*), NULL
FROM (
    SELECT *
    FROM bronze_layer b
    JOIN silver_layer s
        ON b.applicationemployeeid = s.applicationemployeeid
    WHERE
        COALESCE(CAST(b.trainingprovidertypeid AS VARCHAR),'')
        <> COALESCE(CAST(s.trainingprovidertype AS VARCHAR),'')
) t;
