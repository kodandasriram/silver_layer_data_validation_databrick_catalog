---89) dev_iceberg."tmkn-aws-dwh-dev-iceberg-silver".WAGE_BASE

WITH bronze_layer AS (

    SELECT
        COALESCE(
            CAST(w.applicationsupportid AS VARCHAR),
            CAST(t.tws_wageincrementid AS VARCHAR)
        ) AS applicationsupportid,

        w.startdate,
        w.enddate,
        w.newjoblevelid,
        w.newtitle,
        w.newdepartment,
        w.requestedincrementamount,
        w.newwage,
        w.tkshareamt,
        w.incrementpercentage,

        t.tws_name,
        t.tws_current_wage,
        t.tws_required_increment,
        t.tws_tamkeen_share,
        t.tws_new_wage,

        FALSE AS is_deleted

    FROM dev_iceberg."tmkn-aws-dwh-dev-iceberg-bronze".OSUSR_VYW_WAGE w

    LEFT JOIN dev_iceberg."tmkn-aws-dwh-dev-iceberg-bronze".TWS_WAGEINCREMENTBASE t
        ON CAST(w.applicationsupportid AS VARCHAR) = CAST(t.tws_wageincrementid AS VARCHAR)

),

silver_layer AS (

    SELECT
        CAST(applicationsupportid AS VARCHAR) AS applicationsupportid,
        startdate,
        enddate,
        newjoblevelid,
        newtitle,
        newdepartment,
        requestedincrementamount,
        newwage,
        tkshareamt,
        incrementpercentage,

        tws_name,
        tws_current_wage,
        tws_required_increment,
        tws_tamkeen_share,
        tws_new_wage,

        is_deleted

    FROM dev_iceberg."tmkn-aws-dwh-dev-iceberg-silver".WAGE_BASE
)

-- =========================
-- VALIDATION SECTION
-- =========================

-- COUNT CHECK
SELECT 'COUNT_VALIDATION', COUNT(*), (SELECT COUNT(*) FROM silver_layer)
FROM bronze_layer

UNION ALL

-- DUPLICATES IN BRONZE
SELECT 'DUPLICATE_BRONZE', COUNT(*), NULL
FROM (
    SELECT applicationsupportid
    FROM bronze_layer
    GROUP BY applicationsupportid
    HAVING COUNT(*) > 1
)

UNION ALL

-- DUPLICATES IN SILVER
SELECT 'DUPLICATE_SILVER', COUNT(*), NULL
FROM (
    SELECT applicationsupportid
    FROM silver_layer
    GROUP BY applicationsupportid
    HAVING COUNT(*) > 1
)

UNION ALL

-- NULL PK CHECK BRONZE
SELECT 'PK_NULL_BRONZE', COUNT(*), NULL
FROM bronze_layer
WHERE applicationsupportid IS NULL

UNION ALL

-- NULL PK CHECK SILVER
SELECT 'PK_NULL_SILVER', COUNT(*), NULL
FROM silver_layer
WHERE applicationsupportid IS NULL

UNION ALL

-- BRONZE NOT IN SILVER
SELECT 'BRONZE_NOT_IN_SILVER', COUNT(*), NULL
FROM (
    SELECT
        applicationsupportid,
        CAST(newwage AS VARCHAR),
        CAST(tws_name AS VARCHAR)
    FROM bronze_layer

    EXCEPT

    SELECT
        applicationsupportid,
        CAST(newwage AS VARCHAR),
        CAST(tws_name AS VARCHAR)
    FROM silver_layer
)

UNION ALL

-- SILVER NOT IN BRONZE
SELECT 'SILVER_NOT_IN_BRONZE', COUNT(*), NULL
FROM (
    SELECT
        applicationsupportid,
        CAST(newwage AS VARCHAR),
        CAST(tws_name AS VARCHAR)
    FROM silver_layer

    EXCEPT

    SELECT
        applicationsupportid,
        CAST(newwage AS VARCHAR),
        CAST(tws_name AS VARCHAR)
    FROM bronze_layer
)

UNION ALL

-- COLUMN MISMATCH CHECK
SELECT 'COLUMN_MISMATCH_COUNT', COUNT(*), NULL
FROM (
    SELECT *
    FROM bronze_layer b
    JOIN silver_layer s
        ON b.applicationsupportid = s.applicationsupportid
    WHERE
           COALESCE(CAST(b.startdate AS VARCHAR),'') <> COALESCE(CAST(s.startdate AS VARCHAR),'')
        OR COALESCE(CAST(b.enddate AS VARCHAR),'') <> COALESCE(CAST(s.enddate AS VARCHAR),'')
        OR COALESCE(CAST(b.newjoblevelid AS VARCHAR),'') <> COALESCE(CAST(s.newjoblevelid AS VARCHAR),'')
        OR COALESCE(CAST(b.newtitle AS VARCHAR),'') <> COALESCE(CAST(s.newtitle AS VARCHAR),'')
        OR COALESCE(CAST(b.newdepartment AS VARCHAR),'') <> COALESCE(CAST(s.newdepartment AS VARCHAR),'')
        OR COALESCE(CAST(b.requestedincrementamount AS VARCHAR),'') <> COALESCE(CAST(s.requestedincrementamount AS VARCHAR),'')
        OR COALESCE(CAST(b.newwage AS VARCHAR),'') <> COALESCE(CAST(s.newwage AS VARCHAR),'')
        OR COALESCE(CAST(b.tkshareamt AS VARCHAR),'') <> COALESCE(CAST(s.tkshareamt AS VARCHAR),'')
        OR COALESCE(CAST(b.incrementpercentage AS VARCHAR),'') <> COALESCE(CAST(s.incrementpercentage AS VARCHAR),'')
        OR COALESCE(CAST(b.tws_name AS VARCHAR),'') <> COALESCE(CAST(s.tws_name AS VARCHAR),'')
        OR COALESCE(CAST(b.tws_current_wage AS VARCHAR),'') <> COALESCE(CAST(s.tws_current_wage AS VARCHAR),'')
        OR COALESCE(CAST(b.tws_required_increment AS VARCHAR),'') <> COALESCE(CAST(s.tws_required_increment AS VARCHAR),'')
        OR COALESCE(CAST(b.tws_tamkeen_share AS VARCHAR),'') <> COALESCE(CAST(s.tws_tamkeen_share AS VARCHAR),'')
        OR COALESCE(CAST(b.tws_new_wage AS VARCHAR),'') <> COALESCE(CAST(s.tws_new_wage AS VARCHAR),'')
) t;
