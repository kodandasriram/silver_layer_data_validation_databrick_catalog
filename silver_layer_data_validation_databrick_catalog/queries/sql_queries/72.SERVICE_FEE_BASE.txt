WITH bronze_layer AS (
    SELECT
        applicationsupportid,
        programsupporttypesubitemid,
        startdate,
        enddate,
        supportamount,
        isapprovedbyassessor,
        attendingtype,
        actionpoints,
        employercrnumber,
        employercrname,
        details,
        FALSE AS is_deleted
    FROM dev_iceberg."tmkn-aws-dwh-dev-iceberg-bronze".OSUSR_2DA_SERVICEFEE
),

silver_layer AS (
    SELECT
        applicationsupportid,
        programsupporttypesubitemid,
        startdate,
        enddate,
        supportamount,
        isapprovedbyassessor,
        attendingtype,
        actionpoints,
        employercrnumber,
        employercrname,
        details,
        is_deleted
    FROM dev_iceberg."tmkn-aws-dwh-dev-iceberg-silver".SERVICE_FEE_BASE
)

-- =========================
-- VALIDATION OUTPUT
-- =========================

SELECT 'COUNT_VALIDATION' AS validation_type,
       COUNT(*) AS bronze_count,
       (SELECT COUNT(*) FROM silver_layer) AS silver_count
FROM bronze_layer

UNION ALL

SELECT 'DUPLICATE_BRONZE',
       COUNT(*),
       NULL
FROM (
    SELECT applicationsupportid
    FROM bronze_layer
    GROUP BY applicationsupportid
    HAVING COUNT(*) > 1
) t

UNION ALL

SELECT 'DUPLICATE_SILVER',
       COUNT(*),
       NULL
FROM (
    SELECT applicationsupportid
    FROM silver_layer
    GROUP BY applicationsupportid
    HAVING COUNT(*) > 1
) t

UNION ALL

SELECT 'PK_NULL_BRONZE',
       COUNT(*),
       NULL
FROM bronze_layer
WHERE applicationsupportid IS NULL

UNION ALL

SELECT 'PK_NULL_SILVER',
       COUNT(*),
       NULL
FROM silver_layer
WHERE applicationsupportid IS NULL

UNION ALL

SELECT 'BRONZE_NOT_IN_SILVER',
       COUNT(*),
       NULL
FROM (
    SELECT
        CAST(applicationsupportid AS VARCHAR),
        CAST(programsupporttypesubitemid AS VARCHAR),
        CAST(supportamount AS VARCHAR)
    FROM bronze_layer

    EXCEPT

    SELECT
        CAST(applicationsupportid AS VARCHAR),
        CAST(programsupporttypesubitemid AS VARCHAR),
        CAST(supportamount AS VARCHAR)
    FROM silver_layer
) t

UNION ALL

SELECT 'SILVER_NOT_IN_BRONZE',
       COUNT(*),
       NULL
FROM (
    SELECT
        CAST(applicationsupportid AS VARCHAR),
        CAST(programsupporttypesubitemid AS VARCHAR),
        CAST(supportamount AS VARCHAR)
    FROM silver_layer

    EXCEPT

    SELECT
        CAST(applicationsupportid AS VARCHAR),
        CAST(programsupporttypesubitemid AS VARCHAR),
        CAST(supportamount AS VARCHAR)
    FROM bronze_layer
) t

UNION ALL

SELECT 'COLUMN_MISMATCH_COUNT',
       COUNT(*),
       NULL
FROM (
    SELECT b.applicationsupportid
    FROM bronze_layer b
    JOIN silver_layer s
        ON b.applicationsupportid = s.applicationsupportid
    WHERE
           COALESCE(CAST(b.programsupporttypesubitemid AS VARCHAR), '') <> COALESCE(CAST(s.programsupporttypesubitemid AS VARCHAR), '')
        OR COALESCE(CAST(b.startdate AS VARCHAR), '') <> COALESCE(CAST(s.startdate AS VARCHAR), '')
        OR COALESCE(CAST(b.enddate AS VARCHAR), '') <> COALESCE(CAST(s.enddate AS VARCHAR), '')
        OR COALESCE(CAST(b.supportamount AS VARCHAR), '') <> COALESCE(CAST(s.supportamount AS VARCHAR), '')
        OR COALESCE(CAST(b.isapprovedbyassessor AS VARCHAR), '') <> COALESCE(CAST(s.isapprovedbyassessor AS VARCHAR), '')
        OR COALESCE(CAST(b.attendingtype AS VARCHAR), '') <> COALESCE(CAST(s.attendingtype AS VARCHAR), '')
        OR COALESCE(CAST(b.actionpoints AS VARCHAR), '') <> COALESCE(CAST(s.actionpoints AS VARCHAR), '')
        OR COALESCE(CAST(b.employercrnumber AS VARCHAR), '') <> COALESCE(CAST(s.employercrnumber AS VARCHAR), '')
        OR COALESCE(CAST(b.employercrname AS VARCHAR), '') <> COALESCE(CAST(s.employercrname AS VARCHAR), '')
        OR COALESCE(CAST(b.details AS VARCHAR), '') <> COALESCE(CAST(s.details AS VARCHAR), '')
) t;
