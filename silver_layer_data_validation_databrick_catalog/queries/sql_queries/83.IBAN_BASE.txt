WITH bronze_layer AS (
    SELECT
        COALESCE(CAST(s2.id AS VARCHAR), CAST(s1.id AS VARCHAR)) AS id,

        CAST(s2.customerprofileid AS VARCHAR) AS customerprofileid,
        CAST(s2.portaluserid AS VARCHAR) AS portaluserid,

        COALESCE(CAST(s2.bankid AS VARCHAR), CAST(s1.bankid AS VARCHAR)) AS bankid,
        COALESCE(CAST(s2.ibannumber AS VARCHAR), CAST(s1.ibannumber AS VARCHAR)) AS ibannumber,
        COALESCE(CAST(s2.accountname AS VARCHAR), CAST(s1.accountname AS VARCHAR)) AS accountname,

        CAST(s1.bankname AS VARCHAR) AS bankname,
        CAST(s1.ibantypeid AS VARCHAR) AS ibantypeid,

        COALESCE(CAST(s2.ibanstatusid AS VARCHAR), CAST(s1.ibanstatusid AS VARCHAR)) AS ibanstatusid,
        COALESCE(CAST(s2.isdefault AS VARCHAR), CAST(s1.isdefault AS VARCHAR)) AS isdefault,

        CAST(s1.documentid AS VARCHAR) AS documentid,
        CAST(s1.additionalcomment AS VARCHAR) AS additionalcomment,

        CAST(s2.email AS VARCHAR) AS email,
        CAST(s2.mobilenumber AS VARCHAR) AS mobilenumber,
        CAST(s2.mobilecountryprefix AS VARCHAR) AS mobilecountryprefix,
        CAST(s2.isverifiedbytarabut AS VARCHAR) AS isverifiedbytarabut,
        CAST(s2.issalaryiban AS VARCHAR) AS issalaryiban,

        COALESCE(CAST(s2.createdby AS VARCHAR), CAST(s1.createdby AS VARCHAR)) AS createdby,
        CAST(s2.externalbankid AS VARCHAR) AS externalbankid,
        CAST(s2.currency AS VARCHAR) AS currency,

        CAST(s2.deactivatedon AS TIMESTAMP) AS deactivatedon,
        CAST(s2.reasonsfordeactivation AS VARCHAR) AS reasonsfordeactivation,

        COALESCE(CAST(s2.createdon AS TIMESTAMP), CAST(s1.createdon AS TIMESTAMP)) AS createdon,
        COALESCE(CAST(s2.updatedon AS TIMESTAMP), CAST(s1.updatedon AS TIMESTAMP)) AS updatedon,

        FALSE AS is_deleted

    FROM dev_iceberg."tmkn-aws-dwh-dev-iceberg-bronze".OSUSR_PX1_IBAN s1
    FULL OUTER JOIN dev_iceberg."tmkn-aws-dwh-dev-iceberg-bronze".OSUSR_TLV_IBAN s2
        ON CAST(s1.id AS VARCHAR) = CAST(s2.id AS VARCHAR)
),

silver_layer AS (
    SELECT
        CAST(id AS VARCHAR) AS id,
        CAST(customerprofileid AS VARCHAR) AS customerprofileid,
        CAST(portaluserid AS VARCHAR) AS portaluserid,
        CAST(bankid AS VARCHAR) AS bankid,
        CAST(ibannumber AS VARCHAR) AS ibannumber,
        CAST(accountname AS VARCHAR) AS accountname,
        CAST(bankname AS VARCHAR) AS bankname,
        CAST(ibantypeid AS VARCHAR) AS ibantypeid,
        CAST(ibanstatusid AS VARCHAR) AS ibanstatusid,
        CAST(isdefault AS VARCHAR) AS isdefault,
        CAST(documentid AS VARCHAR) AS documentid,
        CAST(additionalcomment AS VARCHAR) AS additionalcomment,
        CAST(email AS VARCHAR) AS email,
        CAST(mobilenumber AS VARCHAR) AS mobilenumber,
        CAST(mobilecountryprefix AS VARCHAR) AS mobilecountryprefix,
        CAST(isverifiedbytarabut AS VARCHAR) AS isverifiedbytarabut,
        CAST(issalaryiban AS VARCHAR) AS issalaryiban,
        CAST(createdby AS VARCHAR) AS createdby,
        CAST(externalbankid AS VARCHAR) AS externalbankid,
        CAST(currency AS VARCHAR) AS currency,
        CAST(deactivatedon AS TIMESTAMP) AS deactivatedon,
        CAST(reasonsfordeactivation AS VARCHAR) AS reasonsfordeactivation,
        CAST(createdon AS TIMESTAMP) AS createdon,
        CAST(updatedon AS TIMESTAMP) AS updatedon,
        is_deleted
    FROM dev_iceberg."tmkn-aws-dwh-dev-iceberg-silver".IBAN_BASE
)

-- =========================
-- VALIDATION
-- =========================

SELECT 'COUNT_VALIDATION' AS validation_type,
       (SELECT COUNT(*) FROM bronze_layer) AS bronze_count,
       (SELECT COUNT(*) FROM silver_layer) AS silver_count

UNION ALL

SELECT 'DUPLICATE_BRONZE',
       COUNT(*),
       NULL
FROM (
    SELECT id
    FROM bronze_layer
    GROUP BY id
    HAVING COUNT(*) > 1
)

UNION ALL

SELECT 'DUPLICATE_SILVER',
       COUNT(*),
       NULL
FROM (
    SELECT id
    FROM silver_layer
    GROUP BY id
    HAVING COUNT(*) > 1
)

UNION ALL

SELECT 'PK_NULL_BRONZE',
       COUNT(*),
       NULL
FROM bronze_layer
WHERE id IS NULL

UNION ALL

SELECT 'PK_NULL_SILVER',
       COUNT(*),
       NULL
FROM silver_layer
WHERE id IS NULL

UNION ALL

-- 🔥 FIXED EXCEPT BLOCK (ALL VARCHAR)
SELECT 'BRONZE_NOT_IN_SILVER',
       COUNT(*),
       NULL
FROM (
    SELECT
        CAST(id AS VARCHAR),
        CAST(ibannumber AS VARCHAR),
        CAST(accountname AS VARCHAR)

UNION ALL

       COUNT(*),

    SELECT
        CAST(id AS VARCHAR),

    FROM silver_layer

    EXCEPT

    SELECT
        CAST(id AS VARCHAR),
        CAST(ibannumber AS VARCHAR),
        CAST(accountname AS VARCHAR)
    FROM bronze_layer
)

UNION ALL

SELECT 'COLUMN_MISMATCH_COUNT',
       COUNT(*),
       NULL
FROM (
    SELECT b.id
    FROM bronze_layer b
    JOIN silver_layer s
        ON b.id = s.id
    WHERE
           COALESCE(b.customerprofileid,'') <> COALESCE(s.customerprofileid,'')
        OR COALESCE(b.portaluserid,'') <> COALESCE(s.portaluserid,'')
        OR COALESCE(b.bankid,'') <> COALESCE(s.bankid,'')
        OR COALESCE(b.ibannumber,'') <> COALESCE(s.ibannumber,'')
        OR COALESCE(b.accountname,'') <> COALESCE(s.accountname,'')
        OR COALESCE(b.ibanstatusid,'') <> COALESCE(s.ibanstatusid,'')
        OR COALESCE(b.isdefault,'') <> COALESCE(s.isdefault,'')
        OR COALESCE(b.currency,'') <> COALESCE(s.currency,'')
) t;
